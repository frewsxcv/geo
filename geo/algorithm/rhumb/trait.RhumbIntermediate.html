let p1 = Point::new(10.0, 20.0);

let i20 = p1.rhumb_intermediate(&p2, 0.2);

assert_relative_eq!(i50.y(), i50_should.y(), epsilon = 0.2);
assert_relative_eq!(i80.x(), i80_should.x(), epsilon = 0.2);
