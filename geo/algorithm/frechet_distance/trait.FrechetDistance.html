use geo::FrechetDistance;
use geo::line_string;

let line_string_a = line_string![
    (x: 1., y: 1.),
    (x: 2., y: 1.),
    (x: 2., y: 2.),
    (x: 3., y: 3.)
];

let line_string_b = line_string![
    (x: 2., y: 2.),
    (x: 0., y: 1.),
    (x: 2., y: 4.),
    (x: 3., y: 4.)
];

let distance = line_string_a.frechet_distance(&line_string_b);

assert_eq!(2., distance);
