7/// Converts geometry to a PostGIS type.
8///
9/// Note that PostGIS databases can include a SRID (spatial reference
10/// system identifier) for geometry stored in them. You should specify
11/// the SRID of your geometry when converting, using `to_postgis_with_srid()`,
12/// or use `to_postgis_wgs84()` if your data is standard WGS84.
13pub trait ToPostgis<T> {
14    /// Converts this geometry to a PostGIS type, using the supplied SRID.
15    fn to_postgis_with_srid(&self, srid: Option<i32>) -> T;
16    /// Converts this WGS84 geometry to a PostGIS type.
17    fn to_postgis_wgs84(&self) -> T {
18        self.to_postgis_with_srid(Some(4326))
19    }
20}

22impl ToPostgis<ewkb::Point> for Coord {
23    fn to_postgis_with_srid(&self, srid: Option<i32>) -> ewkb::Point {
24        ewkb::Point::new(self.x, self.y, srid)
25    }
26}
27
28impl ToPostgis<ewkb::Point> for Point {
29    fn to_postgis_with_srid(&self, srid: Option<i32>) -> ewkb::Point {
30        ewkb::Point::new(self.x(), self.y(), srid)
31    }
32}
33impl ToPostgis<ewkb::LineString> for Line {
34    fn to_postgis_with_srid(&self, srid: Option<i32>) -> ewkb::LineString {
35        let points = vec![
36            self.start_point().to_postgis_with_srid(srid),
37            self.end_point().to_postgis_with_srid(srid),
38        ];
39        ewkb::LineString { points, srid }
40    }
41}
42impl ToPostgis<ewkb::Polygon> for Polygon<f64> {
43    fn to_postgis_with_srid(&self, srid: Option<i32>) -> ewkb::Polygon {
44        let rings = ::std::iter::once(self.exterior())
45            .chain(self.interiors().iter())
46            .map(|x| (*x).to_postgis_with_srid(srid))
47            .collect();
48        ewkb::Polygon { rings, srid }
49    }
50}
51macro_rules! to_postgis_impl {
52    ($from:ident, $to:path, $name:ident) => {
53        impl ToPostgis<$to> for $from<f64> {
54            fn to_postgis_with_srid(&self, srid: Option<i32>) -> $to {
55                let $name = self
56                    .0
57                    .iter()
58                    .map(|x| x.to_postgis_with_srid(srid))
59                    .collect();
60                $to { $name, srid }
61            }
62        }
63    };
64}
65to_postgis_impl!(GeometryCollection, ewkb::GeometryCollection, geometries);
66to_postgis_impl!(MultiPolygon, ewkb::MultiPolygon, polygons);
67to_postgis_impl!(MultiLineString, ewkb::MultiLineString, lines);
68to_postgis_impl!(MultiPoint, ewkb::MultiPoint, points);
69to_postgis_impl!(LineString, ewkb::LineString, points);
70impl ToPostgis<ewkb::Geometry> for Geometry {
71    fn to_postgis_with_srid(&self, srid: Option<i32>) -> ewkb::Geometry {
72        match *self {
73            Geometry::Point(ref p) => ewkb::GeometryT::Point(p.to_postgis_with_srid(srid)),
74            Geometry::Line(ref p) => ewkb::GeometryT::LineString(p.to_postgis_with_srid(srid)),
75            Geometry::LineString(ref p) => {
76                ewkb::GeometryT::LineString(p.to_postgis_with_srid(srid))
77            }
78            Geometry::Polygon(ref p) => ewkb::GeometryT::Polygon(p.to_postgis_with_srid(srid)),
79            Geometry::MultiPoint(ref p) => {
80                ewkb::GeometryT::MultiPoint(p.to_postgis_with_srid(srid))
81            }
82            Geometry::MultiLineString(ref p) => {
83                ewkb::GeometryT::MultiLineString(p.to_postgis_with_srid(srid))
84            }
85            Geometry::MultiPolygon(ref p) => {
86                ewkb::GeometryT::MultiPolygon(p.to_postgis_with_srid(srid))
87            }
88            Geometry::GeometryCollection(ref p) => {
89                ewkb::GeometryT::GeometryCollection(p.to_postgis_with_srid(srid))
90            }
91            Geometry::Rect(ref p) => {
92                ewkb::GeometryT::Polygon(p.to_polygon().to_postgis_with_srid(srid))
93            }
94            Geometry::Triangle(ref p) => {
95                ewkb::GeometryT::Polygon(p.to_polygon().to_postgis_with_srid(srid))
96            }
97        }
98    }
