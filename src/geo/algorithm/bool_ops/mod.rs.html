19/// Boolean Operations on geometry.
20///
21/// Boolean operations are set operations on geometries considered as a subset
22/// of the 2-D plane. The operations supported are: intersection, union,
23/// symmetric difference (xor), and set-difference on pairs of 2-D geometries and
24/// clipping a 1-D geometry with self.
25///
26/// These operations are implemented on [`Polygon`] and the [`MultiPolygon`]
27/// geometries.
28///
29/// # Validity
30///
31/// Note that the operations are strictly well-defined only on *valid*
32/// geometries. However, the implementation generally works well as long as the
33/// interiors of polygons are contained within their corresponding exteriors.
34///
35/// Degenerate 2-d geoms with 0 area are handled, and ignored by the algorithm.
36/// In particular, taking `union` with an empty geom should remove degeneracies
37/// and fix invalid polygons as long the interior-exterior requirement above is
38/// satisfied.
39///
40/// # Performance
41///
42/// For union operations on a large number of [`Polygon`]s or [`MultiPolygon`]s,
43/// using [`unary_union`] will yield far better performance.
44pub trait BooleanOps {
45    type Scalar: BoolOpsNum;
46
47    /// The exterior and interior rings of the geometry.
48    ///
49    /// It doesn't particularly matter which order they are in, as the topology algorithm counts crossings
50    /// to determine the interior and exterior of the polygon.
51    ///
52    /// It is required that the rings are from valid geometries, that the rings not overlap.
53    /// In the case of a MultiPolygon, this requires that none of its polygon's interiors may overlap.
54    fn rings(&self) -> impl Iterator<Item = &LineString<Self::Scalar>>;
55
56    fn boolean_op(
57        &self,
58        other: &impl BooleanOps<Scalar = Self::Scalar>,
59        op: OpType,
60    ) -> MultiPolygon<Self::Scalar> {
61        let subject = self.rings().map(ring_to_shape_path).collect::<Vec<_>>();
62        let clip = other.rings().map(ring_to_shape_path).collect::<Vec<_>>();
63        let shapes = subject.overlay(&clip, op.into(), FillRule::EvenOdd);
64        multi_polygon_from_shapes(shapes)
65    }
66
67    /// Returns the overlapping regions shared by both `self` and `other`.
68    fn intersection(
69        &self,
70        other: &impl BooleanOps<Scalar = Self::Scalar>,
71    ) -> MultiPolygon<Self::Scalar> {
72        self.boolean_op(other, OpType::Intersection)
73    }
74
75    /// Combines the regions of both `self` and `other` into a single geometry, removing
76    /// overlaps and merging boundaries.
77    fn union(&self, other: &impl BooleanOps<Scalar = Self::Scalar>) -> MultiPolygon<Self::Scalar> {
78        self.boolean_op(other, OpType::Union)
79    }
80
81    /// The regions that are in either `self` or `other`, but not in both.
82    fn xor(&self, other: &impl BooleanOps<Scalar = Self::Scalar>) -> MultiPolygon<Self::Scalar> {
83        self.boolean_op(other, OpType::Xor)
84    }
85
86    /// The regions of `self` which are not in `other`.
87    fn difference(
88        &self,
89        other: &impl BooleanOps<Scalar = Self::Scalar>,
90    ) -> MultiPolygon<Self::Scalar> {
91        self.boolean_op(other, OpType::Difference)
92    }
93
94    /// Clip a 1-D geometry with self.
95    ///
96    /// Returns the portion of `ls` that lies within `self` (known as the set-theoeretic
97    /// intersection) if `invert` is false, and the difference (`ls - self`) otherwise.
98    fn clip(
99        &self,
100        multi_line_string: &MultiLineString<Self::Scalar>,
101        invert: bool,
102    ) -> MultiLineString<Self::Scalar> {
103        let subject: Vec<Vec<_>> = multi_line_string
104            .iter()
105            .map(|line_string| line_string.coords().map(|c| BoolOpsCoord(*c)).collect())
106            .collect();
107
108        let clip = self.rings().map(ring_to_shape_path).collect::<Vec<_>>();
109
110        let clip_rule = ClipRule {
111            invert,
112            boundary_included: true,
113        };
114        let paths = subject.clip_by(&clip, FillRule::EvenOdd, clip_rule);
115        i_overlay_integration::convert::multi_line_string_from_paths(paths)
116    }
117}

127/// Efficient [union](BooleanOps::union) of many adjacent / overlapping geometries
128///
129/// This is typically much faster than `union`ing a bunch of geometries together one at a time.
130///
131/// Note: Geometries can be wound in either direction, but the winding order must be consistent,
132/// and each polygon's interior rings must be wound opposite to its exterior.
133///
134/// See [Orient] for more information.
135///
136/// [Orient]: crate::algorithm::orient::Orient
137///
138/// # Arguments
139///
140/// `boppables`: A collection of `Polygon` or `MultiPolygons` to union together.
141///
142/// returns the union of all the inputs.
143///
144/// # Examples
145///
146/// ```
147/// use geo::algorithm::unary_union;
148/// use geo::wkt;
149///
150/// let right_piece = wkt!(POLYGON((4. 0.,4. 4.,8. 4.,8. 0.,4. 0.)));
151/// let left_piece = wkt!(POLYGON((0. 0.,0. 4.,4. 4.,4. 0.,0. 0.)));
152///
153/// // touches neither right nor left piece
154/// let separate_piece = wkt!(POLYGON((14. 10.,14. 14.,18. 14.,18. 10.,14. 10.)));
155///
156/// let polygons = vec![left_piece, separate_piece, right_piece];
157/// let actual_output = unary_union(&polygons);
158///
159/// let expected_output = wkt!(MULTIPOLYGON(
160///     // left and right piece have been combined
161///     ((0.0 0.0, 8.0 0.0, 8.0 4.0, 0.0 4.0, 0.0 0.0)),
162///     // separate piece remains separate
163///     ((14.0 10.0, 18.0 10.0, 18.0 14.0, 14.0 14.0, 14.0 10.0))
164/// ));
165/// assert_eq!(actual_output, expected_output);
166/// ```
167pub fn unary_union<'a, B: BooleanOps + 'a>(
168    boppables: impl IntoIterator<Item = &'a B>,
169) -> MultiPolygon<B::Scalar> {
170    let mut winding_order: Option<WindingOrder> = None;
171    let subject = boppables
172        .into_iter()
173        .flat_map(|boppable| {
174            let rings = boppable.rings();
175            rings
176                .map(|ring| {
177                    if winding_order.is_none() {
178                        winding_order = ring.winding_order();
179                    }
180                    ring_to_shape_path(ring)
181                })
182                .collect::<Vec<_>>()
183        })
184        .collect::<Vec<_>>();
185
186    let fill_rule = if winding_order == Some(WindingOrder::Clockwise) {
187        FillRule::Positive
188    } else {
189        FillRule::Negative
190    };
191
192    let shapes = FloatOverlay::with_subj(&subject).overlay(OverlayRule::Subject, fill_rule);
193    multi_polygon_from_shapes(shapes)
194}

196impl<T: BoolOpsNum> BooleanOps for Polygon<T> {
197    type Scalar = T;
198
199    fn rings(&self) -> impl Iterator<Item = &LineString<Self::Scalar>> {
200        std::iter::once(self.exterior()).chain(self.interiors())
201    }
202}

204impl<T: BoolOpsNum> BooleanOps for MultiPolygon<T> {
205    type Scalar = T;
206
207    fn rings(&self) -> impl Iterator<Item = &LineString<Self::Scalar>> {
208        self.iter().flat_map(BooleanOps::rings)
209    }
