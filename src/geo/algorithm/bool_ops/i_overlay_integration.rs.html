6/// A geometry coordinate scalar suitable for performing geometric boolean operations.
7pub trait BoolOpsNum: GeoNum + FloatNumber {}
8impl<T: GeoNum + FloatNumber> BoolOpsNum for T {}
9
10/// New type for `Coord` that implements `FloatPointCompatible` for `BoolOpsNum` to
11/// circumvent orphan rule, since Coord is defined in geo_types.
12#[derive(Copy, Clone, Debug)]
13pub struct BoolOpsCoord<T: BoolOpsNum>(pub(crate) Coord<T>);
14
15impl<T: BoolOpsNum> FloatPointCompatible<T> for BoolOpsCoord<T> {
16    fn from_xy(x: T, y: T) -> Self {
17        Self(Coord { x, y })
18    }
19
20    fn x(&self) -> T {
21        self.0.x
22    }
23
24    fn y(&self) -> T {
25        self.0.y
26    }
27}

36    pub fn line_string_from_path<T: BoolOpsNum>(path: Vec<BoolOpsCoord<T>>) -> LineString<T> {
37        let coords = path.into_iter().map(|bops_coord| bops_coord.0).collect();
38        LineString(coords)
39    }
40
41    pub fn multi_line_string_from_paths<T: BoolOpsNum>(
42        paths: Vec<Vec<BoolOpsCoord<T>>>,
43    ) -> MultiLineString<T> {
44        let line_strings = paths.into_iter().map(|p| line_string_from_path(p));
45        MultiLineString(line_strings.collect())
46    }
47
48    pub fn polygon_from_shape<T: BoolOpsNum>(shape: Vec<Vec<BoolOpsCoord<T>>>) -> Polygon<T> {
49        let mut rings = shape.into_iter().map(|path| {
50            // From i_overlay: > Note: Outer boundary paths have a clockwise order, and holes have a counterclockwise order.
51            // Which is the opposite convention we use.
52            let mut line_string = line_string_from_path(path);
53            line_string.close();
54            line_string.0.reverse();
55            line_string
56        });
57        let exterior = rings.next().unwrap_or(LineString::new(vec![]));
58
59        Polygon::new(exterior, rings.collect())
60    }
61
62    pub fn multi_polygon_from_shapes<T: BoolOpsNum>(
63        shapes: Vec<Vec<Vec<BoolOpsCoord<T>>>>,
64    ) -> MultiPolygon<T> {
65        let polygons = shapes.into_iter().map(|s| polygon_from_shape(s));
66        MultiPolygon(polygons.collect())
67    }
68
69    pub fn ring_to_shape_path<T: BoolOpsNum>(line_string: &LineString<T>) -> Vec<BoolOpsCoord<T>> {
70        if line_string.0.is_empty() {
71            return vec![];
72        }
73        // In geo, Polygon rings are explicitly closed LineStrings â€” their final coordinate is the same as their first coordinate,
74        // however in i_overlay, shape paths are implicitly closed, so we skip the last coordinate.
75        let coords = &line_string.0[..line_string.0.len() - 1];
76        coords.iter().copied().map(BoolOpsCoord).collect()
77    }

79    impl From<OpType> for OverlayRule {
80        fn from(op: OpType) -> Self {
81            match op {
82                OpType::Intersection => OverlayRule::Intersect,
83                OpType::Union => OverlayRule::Union,
84                OpType::Difference => OverlayRule::Difference,
85                OpType::Xor => OverlayRule::Xor,
86            }
87        }

100    #[test]
101    // see https://github.com/georust/geo/issues/1309
102    fn test_winding_order() {
103        let poly1 = polygon!((x: 0.0, y: 0.0), (x: 1.0, y: 0.0), (x: 1.0, y: 1.0));
104        assert!(matches!(
105            poly1.exterior().winding_order(),
106            Some(WindingOrder::CounterClockwise)
107        ));
108
109        {
110            let union = poly1.union(&polygon!());
111            assert_eq!(union.0.len(), 1);
112
113            let union = &union.0[0];
114            assert!(matches!(
115                union.exterior().winding_order(),
116                Some(WindingOrder::CounterClockwise)
117            ));
118        }
119        {
120            let intersection = poly1.intersection(&poly1);
121            assert_eq!(intersection.0.len(), 1);
122
123            let intersection = &intersection.0[0];
124            assert!(matches!(
125                intersection.exterior().winding_order(),
126                Some(WindingOrder::CounterClockwise)
127            ));
128        }
129
130        let poly2 = polygon!((x: 0.0, y: 0.0), (x: 1.0, y: 1.0), (x: 0.0, y: 1.0));
131        assert!(matches!(
132            poly2.exterior().winding_order(),
133            Some(WindingOrder::CounterClockwise)
134        ));
135
136        {
137            let union = poly1.union(&poly2);
138            assert_eq!(union.0.len(), 1);
139
140            let union = &union.0[0];
141            assert!(union.interiors().is_empty());
142            assert!(matches!(
143                union.exterior().winding_order(),
144                Some(WindingOrder::CounterClockwise)
145            ));
146        }
147    }
