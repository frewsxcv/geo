9/// Either a line segment or a point.
10///
11/// The coordinates are ordered (see [`SweepPoint`]) and a line
12/// segment must have distinct points (use the `Point` variant if the
13/// coordinates are the equal).
14#[derive(Clone, Copy)]
15pub enum LineOrPoint<T: GeoNum> {
16    Point(SweepPoint<T>),
17    Line {
18        left: SweepPoint<T>,
19        right: SweepPoint<T>,
20    },
21}
22
23impl<T: GeoNum> std::fmt::Debug for LineOrPoint<T> {
24    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
25        match self {
26            LineOrPoint::Point(p) => f.debug_tuple("Pt").field(&p.x_y()).finish(),
27            LineOrPoint::Line { left, right } => f
28                .debug_tuple("LPt")
29                .field(&left.x_y())
30                .field(&right.x_y())
31                .finish(),
32        }
33    }
34}

42impl<T: GeoNum> From<(SweepPoint<T>, SweepPoint<T>)> for LineOrPoint<T> {
43    fn from((start, end): (SweepPoint<T>, SweepPoint<T>)) -> Self {
44        match start.cmp(&end) {
45            Ordering::Less => Self::Line {
46                left: start,
47                right: end,
48            },
49            Ordering::Equal => Self::Point(start),
50            Ordering::Greater => Self::Line {
51                left: end,
52                right: start,
53            },
54        }
55    }
56}

74impl<T: GeoNum> LineOrPoint<T> {
75    /// Checks if the variant is a line.
76    #[inline]
77    pub fn is_line(&self) -> bool {
78        matches!(self, Self::Line { .. })
79    }
80
81    /// Return a [`Line`] representation of self.
82    #[inline]
83    pub fn line(&self) -> Line<T> {
84        match self {
85            LineOrPoint::Point(p) => Line::new(**p, **p),
86            LineOrPoint::Line { left, right } => Line::new(**left, **right),
87        }
88    }
89
90    #[inline]
91    pub fn left(&self) -> SweepPoint<T> {
92        match self {
93            LineOrPoint::Point(p) => *p,
94            LineOrPoint::Line { left, .. } => *left,
95        }
96    }
97
98    #[inline]
99    pub fn right(&self) -> SweepPoint<T> {
100        match self {
101            LineOrPoint::Point(p) => *p,
102            LineOrPoint::Line { right, .. } => *right,
103        }
104    }
105
106    #[cfg(test)]
107    pub fn coords_equal(&self, other: &LineOrPoint<T>) -> bool {
108        self.is_line() == other.is_line() && self.end_points() == other.end_points()
109    }
110
111    #[inline]
112    pub fn end_points(&self) -> (SweepPoint<T>, SweepPoint<T>) {
113        match self {
114            LineOrPoint::Point(p) => (*p, *p),
115            LineOrPoint::Line { left, right } => (*left, *right),
116        }
117    }
118
119    pub fn new(left: SweepPoint<T>, right: SweepPoint<T>) -> Self {
120        if left == right {
121            Self::Point(left)
122        } else {
123            Self::Line { left, right }
124        }
125    }
126
127    pub fn orient2d(&self, other: Coord<T>) -> Orientation {
128        let (left, right) = match self {
129            LineOrPoint::Point(p) => (**p, **p),
130            LineOrPoint::Line { left, right } => (**left, **right),
131        };
132        T::Ker::orient2d(left, right, other)
133    }
134}

136/// Equality based on ordering defined for segments as per algorithm.
137impl<T: GeoNum> PartialEq for LineOrPoint<T> {
138    #[inline]
139    fn eq(&self, other: &Self) -> bool {
140        self.partial_cmp(other) == Some(Ordering::Equal)
141    }
142}
143
144/// Ordering defined for segments as per algorithm.
145///
146/// Requires the following conditions:
147///
148/// 1. If comparing two lines, both the left ends must be strictly
149///    smaller than both right ends.
150///
151/// 2. A point is treated as a infinitesimal small vertical segment
152///    centered at its coordinates.
153impl<T: GeoNum> PartialOrd for LineOrPoint<T> {
154    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
155        match (self, other) {
156            (LineOrPoint::Point(p), LineOrPoint::Point(o)) => {
157                if p == o {
158                    Some(Ordering::Equal)
159                } else {
160                    // Unequal points do not satisfy pre-condition and
161                    // can't be ordered.
162                    None
163                }
164            }
165            (LineOrPoint::Point(_), LineOrPoint::Line { .. }) => {
166                other.partial_cmp(self).map(Ordering::reverse)
167            }
168            (LineOrPoint::Line { left, right }, LineOrPoint::Point(p)) => {
169                if p > right || left > p {
170                    return None;
171                }
172                Some(
173                    T::Ker::orient2d(**left, **right, **p)
174                        .as_ordering()
175                        .then(Ordering::Greater),
176                )
177            }
178            (
179                LineOrPoint::Line {
180                    left: left_a,
181                    right: right_a,
182                },
183                LineOrPoint::Line {
184                    left: left_b,
185                    right: right_b,
186                },
187            ) => {
188                if left_a > left_b {
189                    return other.partial_cmp(self).map(Ordering::reverse);
190                }
191                if left_a >= right_b || left_b >= right_a {
192                    return None;
193                }
194
195                // Assertion: p1 <= p2
196                // Assertion: pi < q_j
197                Some(
198                    T::Ker::orient2d(**left_a, **right_a, **left_b)
199                        .as_ordering()
200                        .then_with(|| {
201                            T::Ker::orient2d(**left_a, **right_a, **right_b).as_ordering()
202                        }),
203                )
204            }
205        }
206    }
207}

209impl<T: GeoFloat> LineOrPoint<T> {
210    /// Intersect a line with self and return a point, a overlapping segment or `None`.
211    ///
212    /// The `other` argument must be a line variant (debug builds will panic otherwise).
213    pub fn intersect_line(&self, other: &Self) -> Option<Self> {
214        debug_assert!(other.is_line(), "tried to intersect with a point variant!");
215
216        let line = other.line();
217        match self {
218            LineOrPoint::Point(p) => {
219                use crate::Intersects;
220                if line.intersects(&**p) {
221                    Some(*self)
222                } else {
223                    None
224                }
225            }
226            LineOrPoint::Line { left, right } => {
227                line_intersection(self.line(), line).map(|l| match l {
228                    LineIntersection::SinglePoint {
229                        intersection,
230                        is_proper,
231                    } => {
232                        let mut pt = intersection;
233                        if is_proper && (&pt == left.deref()) {
234                            if left.x == right.x {
235                                pt.y = pt.y.next_after(T::infinity());
236                            } else {
237                                pt.x = pt.x.next_after(T::infinity());
238                            }
239                        }
240                        pt.into()
241                    }
242                    LineIntersection::Collinear { intersection } => intersection.into(),
243                })
244            }
245        }
246    }
247
248    pub fn intersect_line_ordered(&self, other: &Self) -> Option<Self> {
249        let ord = self.partial_cmp(other);
250        match self.intersect_line(other) {
251            Some(Self::Point(p)) => {
252                // NOTE: A key issue with using non-exact numbers (f64, etc.) in
253                // this algo. is that line-intersection may return
254                // counter-intuitive points.
255                //
256                // Specifically, this causes two issues:
257                //
258                // 1. The point of intersection r lies between the end-points in
259                // the lexicographic ordering. However, with finite repr., the
260                // line (1, 1) - (1 + eps, -1), where eps is ulp(1), does not
261                // admit r that lies between the end-points. Further, the
262                // end-points may be a very bad approximation to the actual
263                // intersection points (eg. intersect with x-axis).
264                //
265                // We detect and force r to be greater than both end-points; the
266                // other case is not easy to handle as the sweep has already
267                // progressed to a p strictly > r already.
268                //
269                // 2. The more severe issue is that in general r may not lie
270                // exactly on the line. Thus, with the segment stored on the
271                // active-segments tree (B-Tree / Splay), this may in adverse
272                // cases, cause the ordering between the segments to be
273                // incorrect, hence invalidating the segments. This is not easy
274                // to correct without a intrusive data-structure built
275                // specifically for this algo., that can track the neighbors of
276                // tree-nodes, and fix / report this issue. The crate
277                // `btree-slab` seems like a great starting point.
278                let (mut x, y) = p.x_y();
279
280                let c = self.left();
281                if x == c.x && y < c.y {
282                    x = x.next_after(T::infinity());
283                }
284
285                let p = Coord { x, y }.into();
286                debug_assert!(
287                    p >= self.left(),
288                    "line intersection before first line: {p:?}\n\tLine({lp1:?} - {lp2:?}) X Line({lp3:?} - {lp4:?})",
289                    lp1 = self.left(),
290                    lp2 = self.right(),
291                    lp3 = other.left(),
292                    lp4 = other.right(),
293                );
294                debug_assert!(
295                    p >= other.left(),
296                    "line intersection before second line: {p:?}\n\tLine({lp1:?} - {lp2:?}) X Line({lp3:?} - {lp4:?})",
297                    lp1 = self.left(),
298                    lp2 = self.right(),
299                    lp3 = other.left(),
300                    lp4 = other.right(),
301                );
302
303                if let Some(ord) = ord {
304                    let l1 = LineOrPoint::from((self.left(), p));
305                    let l2 = LineOrPoint::from((other.left(), p));
306                    let cmp = l1.partial_cmp(&l2).unwrap();
307                    if l1.is_line() && l2.is_line() && cmp.then(ord) != ord {
308                        debug!(
309                            "ordering changed by intersection: {l1:?} {ord:?} {l2:?}",
310                            l1 = self,
311                            l2 = other
312                        );
313                        debug!("\tparts: {l1:?}, {l2:?}");
314                        debug!("\tintersection: {p:?} {cmp:?}");
315
316                        // RM: This is a complicated intersection that is changing the ordering.
317                        // Heuristic: approximate with a trivial intersection point that preserves the topology.
318                        return Some(if self.left() > other.left() {
319                            self.left().into()
320                        } else {
321                            other.left().into()
322                        });
323                    }
324                }
325                Some(Self::Point(p))
326            }
327            e => e,
328        }
329    }
