9/// Determine the similarity between two `LineStrings` using the [Frechet distance].
10///
11/// Based on [Computing Discrete Frechet Distance] by T. Eiter and H. Mannila.
12///
13/// [Frechet distance]: https://en.wikipedia.org/wiki/Fr%C3%A9chet_distance
14/// [Computing Discrete Frechet Distance]: http://www.kr.tuwien.ac.at/staff/eiter/et-archive/cdtr9464.pdf
15pub trait FrechetDistance<T, Rhs = Self> {
16    /// Determine the similarity between two `LineStrings` using the [Frechet distance].
17    ///
18    /// # Examples
19    ///
20    /// ```
21    /// use geo::FrechetDistance;
22    /// use geo::line_string;
23    ///
24    /// let line_string_a = line_string![
25    ///     (x: 1., y: 1.),
26    ///     (x: 2., y: 1.),
27    ///     (x: 2., y: 2.),
28    ///     (x: 3., y: 3.)
29    /// ];
30    ///
31    /// let line_string_b = line_string![
32    ///     (x: 2., y: 2.),
33    ///     (x: 0., y: 1.),
34    ///     (x: 2., y: 4.),
35    ///     (x: 3., y: 4.)
36    /// ];
37    ///
38    /// let distance = line_string_a.frechet_distance(&line_string_b);
39    ///
40    /// assert_eq!(2., distance);
41    /// ```
42    ///
43    /// [Frechet distance]: https://en.wikipedia.org/wiki/Fr%C3%A9chet_distance
44    fn frechet_distance(&self, rhs: &Rhs) -> T;
45}
