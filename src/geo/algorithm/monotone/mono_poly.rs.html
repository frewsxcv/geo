8/// Monotone polygon
9///
10/// A monotone polygon is a polygon that can be decomposed into two monotone
11/// chains (along the X-axis). This implies any vertical line intersects the
12/// polygon at most twice (or not at all).  These polygons support
13/// point-in-polygon queries in `O(log n)` time; use the `Intersects<Coord>`
14/// trait to query.
15///
16/// This structure cannot be directly constructed.  Use
17/// `crate::algorithm::monotone_subdivision` algorithm to obtain a
18/// `Vec<MonoPoly>`.  Consider using `MonotonicPolygons` instead if you are not
19/// interested in the individual monotone polygons.
20#[derive(Clone, PartialEq)]
21pub struct MonoPoly<T: GeoNum> {
22    top: LineString<T>,
23    bot: LineString<T>,
24    bounds: Rect<T>,
25}
26
27impl<T: GeoNum> BoundingRect<T> for MonoPoly<T> {
28    type Output = Rect<T>;
29
30    fn bounding_rect(&self) -> Self::Output {
31        self.bounds
32    }
33}
34impl<T: GeoNum> std::fmt::Debug for MonoPoly<T> {
35    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
36        let top: Vec<SweepPoint<T>> = self.top.0.iter().map(|c| (*c).into()).collect();
37        let bot: Vec<SweepPoint<T>> = self.bot.0.iter().map(|c| (*c).into()).collect();
38        f.debug_struct("MonoPoly")
39            .field("top", &top)
40            .field("bot", &bot)
41            .finish()
42    }

45impl<T: GeoNum> MonoPoly<T> {
46    /// Create a monotone polygon from the top and bottom chains.
47    ///
48    /// Note: each chain must be a strictly increasing sequence (in the lexicographic
49    /// order), with the same start and end points.  Further, the top chain must be
50    /// strictly above the bottom chain except at the end-points.  Not all these
51    /// conditions are checked, and the algorithm may panic if they are not met.
52    pub(super) fn new(top: LineString<T>, bot: LineString<T>) -> Self {
53        debug_assert_eq!(top.0.first(), bot.0.first());
54        debug_assert_eq!(top.0.last(), bot.0.last());
55        debug_assert_ne!(top.0.first(), top.0.last());
56        let bounds = get_bounding_rect(top.0.iter().chain(bot.0.iter())).unwrap();
57        Self { top, bot, bounds }
58    }
59
60    /// Get a reference to the mono poly's top chain.
61    #[must_use]
62    pub fn top(&self) -> &LineString<T> {
63        &self.top
64    }
65
66    /// Get a reference to the mono poly's bottom chain.
67    #[must_use]
68    pub fn bot(&self) -> &LineString<T> {
69        &self.bot
70    }
71
72    /// Convert self to (top, bottom) pair of chains.
73    pub fn into_ls_pair(self) -> (LineString<T>, LineString<T>) {
74        (self.top, self.bot)
75    }
76
77    /// Get the pair of segments in the chain that intersects the line parallel
78    /// to the Y-axis at the given x-coordinate.  Ties are broken by picking the
79    /// segment with lower index, i.e. the segment closer to the start of the
80    /// chains.
81    pub fn bounding_segment(&self, x: T) -> Option<(Line<T>, Line<T>)> {
82        // binary search for the segment that contains the x coordinate.
83        let tl_idx = self.top.0.partition_point(|c| c.x < x);
84        if tl_idx == 0 && self.top.0[0].x != x {
85            return None;
86        }
87        let bl_idx = self.bot.0.partition_point(|c| c.x < x);
88        if bl_idx == 0 {
89            debug_assert_eq!(tl_idx, 0);
90            debug_assert_eq!(self.bot.0[0].x, x);
91            return Some((
92                Line::new(self.top.0[0], self.top.0[1]),
93                Line::new(self.bot.0[0], self.bot.0[1]),
94            ));
95        } else {
96            debug_assert_ne!(tl_idx, 0);
97        }
98
99        Some((
100            Line::new(self.top.0[tl_idx - 1], self.top.0[tl_idx]),
101            Line::new(self.bot.0[bl_idx - 1], self.bot.0[bl_idx]),
102        ))
103    }
104
105    /// Convert self into a [`Polygon`].
106    pub fn into_polygon(self) -> Polygon<T> {
107        let mut down = self.bot.0;
108        let mut top = self.top.0;
109
110        down.reverse();
111        assert_eq!(down.first(), top.last());
112        top.extend(down.drain(1..));
113
114        let geom = LineString(top);
115        debug_assert!(geom.is_closed());
116
117        Polygon::new(geom, vec![])
118    }
119}

121impl<T: GeoNum> CoordinatePosition for MonoPoly<T> {
122    type Scalar = T;
123
124    fn calculate_coordinate_position(
125        &self,
126        coord: &Coord<Self::Scalar>,
127        is_inside: &mut bool,
128        boundary_count: &mut usize,
129    ) {
130        if !self.bounds.intersects(coord) {
131            return;
132        }
133        let (top, bot) = if let Some(t) = self.bounding_segment(coord.x) {
134            t
135        } else {
136            return;
137        };
138
139        match T::Ker::orient2d(top.start, *coord, top.end) {
140            Orientation::Clockwise => return,
141            Orientation::Collinear => {
142                *is_inside = true;
143                *boundary_count += 1;
144                return;
145            }
146            _ => {}
147        }
148        match T::Ker::orient2d(bot.start, *coord, bot.end) {
149            Orientation::CounterClockwise => (),
150            Orientation::Collinear => {
151                *is_inside = true;
152                *boundary_count += 1;
153            }
154            _ => {
155                *is_inside = true;
156            }
157        }
158    }
159}
160impl<T: GeoNum> Intersects<Coord<T>> for MonoPoly<T> {
161    fn intersects(&self, other: &Coord<T>) -> bool {
162        self.coordinate_position(other) != CoordPos::Outside
163    }
