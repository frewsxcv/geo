7/// Simple planar sweep algorithm.
8///
9/// Performs a planar sweep along the x-axis on a set of line or points.  This
10/// can be initialized with a iterator of line segments or points, along with an
11/// optional payload.
12///
13/// The sweep is used to:
14///
15/// - iterate over all end-points of the input line or points in lex. order
16///
17/// - query the set of active segments at the current iteration point: these are
18///   the segments currently intersecting the sweep line, and are ordered by their
19///    position on the line
20///
21/// # Note
22///
23/// This is a simpler version, which does not support intersections that are
24/// interior to both segments.  That is, every intersection between two segments
25/// should be at the end point of at least one of them.  In particular, overlaps
26/// are also not supported (will panic).
27pub(crate) struct SimpleSweep<T: GeoNum, P: Debug> {
28    events: BinaryHeap<Event<T, RcSegment<T, P>>>,
29    active_segments: VecSet<Active<RcSegment<T, P>>>,
30}
31
32impl<T: GeoNum, P: Debug + Clone> SimpleSweep<T, P> {
33    pub(crate) fn new<I, D>(iter: I) -> Self
34    where
35        I: IntoIterator<Item = D>,
36        D: Into<Segment<T, P>>,
37    {
38        let iter = iter.into_iter();
39        let size = {
40            let (min_size, max_size) = iter.size_hint();
41            max_size.unwrap_or(min_size)
42        };
43        let mut events = BinaryHeap::with_capacity(size);
44        let active_segments = VecSet::default();
45
46        for cr in iter {
47            let segment = RcSegment::from(cr.into());
48            events.extend(segment.events());
49        }
50
51        SimpleSweep {
52            events,
53            active_segments,
54        }
55    }
56
57    /// Progress and obtain the next sweep point along with the set of segments
58    /// ending at the current sweep point.
59    ///
60    /// The segments are returned as per the ordering of their `EventType`; in
61    /// particular, all segments ending at the current sweep point are returned
62    /// before the ones starting at the current sweep point.  The idx of the
63    /// first segment starting at the current sweep point is returned in the
64    /// `split_idx` parameter.
65    pub(crate) fn next_point<F: FnMut(RcSegment<T, P>, EventType)>(
66        &mut self,
67        mut f: F,
68    ) -> Option<SweepPoint<T>> {
69        let point = self.peek_point();
70        while let Some(pt) = point {
71            let ev = self.events.pop().unwrap();
72            self.handle_event(ev, &mut |ev| {
73                let segment = ev.payload;
74                let ty = ev.ty;
75                f(segment, ty);
76            });
77            if self.peek_point() != Some(pt) {
78                break;
79            }
80        }
81        point
82    }
83
84    fn handle_event<F>(&mut self, event: Event<T, RcSegment<T, P>>, cb: &mut F)
85    where
86        F: FnMut(Event<T, RcSegment<T, P>>),
87    {
88        // We may get spurious events from adjusting the line segment.  Ignore.
89        if event.point != event.payload.line().left() && event.point != event.payload.line().right()
90        {
91            return;
92        }
93
94        use EventType::*;
95        let segment = &event.payload;
96        trace!(
97            "handling event: {pt:?} ({ty:?}) @ {seg:?}",
98            pt = event.point,
99            ty = event.ty,
100            seg = segment,
101        );
102
103        match &event.ty {
104            LineLeft => {
105                let mut idx = self.active_segments.index_not_of(segment);
106                for is_next in [false, true] {
107                    let (active, split) = if !is_next {
108                        if idx > 0 {
109                            let active = &self.active_segments[idx - 1];
110                            (active, self.check_interior_intersection(active, segment))
111                        } else {
112                            continue;
113                        }
114                    } else if idx < self.active_segments.len() {
115                        let active = &self.active_segments[idx];
116                        (active, self.check_interior_intersection(active, segment))
117                    } else {
118                        continue;
119                    };
120
121                    match split {
122                        SplitResult::SplitA(pt) => {
123                            let new_seg = active.split_at(pt);
124                            let [_, ev] = active.events();
125                            self.events.push(ev);
126                            self.events.extend(new_seg.events());
127                        }
128                        SplitResult::SplitB(pt) => {
129                            let new_seg = segment.split_at(pt);
130                            let [_, ev] = segment.events();
131                            self.events.push(ev);
132                            self.events.extend(new_seg.events());
133                        }
134                        SplitResult::None => {}
135                    }
136
137                    // Special case:  if we split at the current event point, then
138                    // we have LineRight events in the queue that have to be
139                    // processed before this.
140
141                    // There's always a top as this is a left event.
142                    while self.events.peek().unwrap() > &event {
143                        debug_assert_eq!(self.events.peek().unwrap().ty, LineRight);
144                        debug_assert_eq!(self.events.peek().unwrap().point, event.point);
145
146                        let ev = self.events.pop().unwrap();
147                        self.handle_event(ev, cb);
148                        if !is_next {
149                            idx -= 1;
150                        }
151                    }
152                }
153
154                self.active_segments.insert_at(idx, segment.clone());
155            }
156            LineRight => {
157                let idx = self.active_segments.index_of(segment);
158                self.active_segments.remove_at(idx);
159
160                if idx > 0 && idx < self.active_segments.len() {
161                    let prev = &self.active_segments[idx - 1];
162                    let next = &self.active_segments[idx];
163                    self.check_interior_intersection(prev, next);
164                }
165            }
166            _ => {}
167        }
168        cb(event);
169    }
170
171    #[inline]
172    pub(super) fn peek_point(&self) -> Option<SweepPoint<T>> {
173        self.events.peek().map(|e| e.point)
174    }
175
176    pub(super) fn prev_active_from_geom(&self, geom: LineOrPoint<T>) -> Option<RcSegment<T, P>> {
177        let part_idx = self.active_segments.partition_point(|s| s.line() < geom);
178        if part_idx == 0 {
179            None
180        } else {
181            Some(self.active_segments[part_idx - 1].0.clone())
182        }
183    }
184
185    /// Check if the two segments intersect at a point interior to one of them.
186    fn check_interior_intersection(
187        &self,
188        a: &RcSegment<T, P>,
189        b: &RcSegment<T, P>,
190    ) -> SplitResult<T> {
191        let la = a.line();
192        let lb = b.line();
193
194        let lal = la.left();
195        let lar = la.right();
196
197        let lbl = lb.left();
198        let lbr = lb.right();
199
200        if lal < lbl && lbl < lar && la.orient2d(*lbl) == Orientation::Collinear {
201            SplitResult::SplitA(lbl)
202        } else if lal < lbr && lbr < lar && la.orient2d(*lbr) == Orientation::Collinear {
203            SplitResult::SplitA(lbr)
204        } else if lbl < lal && lal < lbr && lb.orient2d(*lal) == Orientation::Collinear {
205            SplitResult::SplitB(lal)
206        } else if lbl < lar && lar < lbr && lb.orient2d(*lar) == Orientation::Collinear {
207            SplitResult::SplitB(lar)
208        } else {
209            SplitResult::None
210        }
211    }
212}
213
214enum SplitResult<T: GeoNum> {
215    SplitA(SweepPoint<T>),
216    SplitB(SweepPoint<T>),
217    None,
