4pub(crate) fn check_coord_is_not_finite<T: CoordFloat>(geom: &Coord<T>) -> bool {
5    if geom.x.is_finite() && geom.y.is_finite() {
6        return false;
7    }
8    true
9}

11pub(crate) fn robust_check_points_are_collinear<T: CoordFloat>(
12    p0: &Coord<T>,
13    p1: &Coord<T>,
14    p2: &Coord<T>,
15) -> bool {
16    orient2d(
17        RobustCoord {
18            x: p0.x.to_f64().unwrap(),
19            y: p0.y.to_f64().unwrap(),
20        },
21        RobustCoord {
22            x: p1.x.to_f64().unwrap(),
23            y: p1.y.to_f64().unwrap(),
24        },
25        RobustCoord {
26            x: p2.x.to_f64().unwrap(),
27            y: p2.y.to_f64().unwrap(),
28        },
29    ) == 0.
30}

32pub(crate) fn check_too_few_points<T: CoordFloat>(geom: &LineString<T>, is_ring: bool) -> bool {
33    let n_pts = if is_ring { 4 } else { 2 };
34    if geom.remove_repeated_points().0.len() < n_pts {
35        return true;
36    }
37    false
38}

40pub(crate) fn linestring_has_self_intersection<F: GeoFloat>(geom: &LineString<F>) -> bool {
41    // This need more test to see if we detect "spikes" correctly.
42    // Maybe we could also use https://docs.rs/geo/latest/geo/algorithm/line_intersection/fn.line_intersection.html
43    // to compute the intersection, see if it is a single point or not, etc.
44    for (i, line) in geom.lines().enumerate() {
45        for (j, other_line) in geom.lines().enumerate() {
46            if i != j
47                && line.intersects(&other_line)
48                && line.start != other_line.end
49                && line.end != other_line.start
50            {
51                return true;
52            }
53        }
54    }
55    false
56}
