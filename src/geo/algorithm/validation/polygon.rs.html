8/// A [`Polygon`] must follow these rules to be valid:
9/// - [x] the polygon boundary rings (the exterior shell ring and interior hole rings) are simple (do not cross or self-touch). Because of this a polygon cannnot have cut lines, spikes or loops. This implies that polygon holes must be represented as interior rings, rather than by the exterior ring self-touching (a so-called "inverted hole").
10/// - [x] boundary rings do not cross
11/// - [x] boundary rings may touch at points but only as a tangent (i.e. not in a line)
12/// - [x] interior rings are contained in the exterior ring
13/// - [ ] the polygon interior is simply connected (i.e. the rings must not touch in a way that splits the polygon into more than one part)
14///
15/// Note: the simple connectivity of the interior is not checked by this implementation.
16#[derive(Debug, Clone, PartialEq)]
17pub enum InvalidPolygon {
18    /// A ring must have at least 4 points to be valid. Note that, in order to close the ring, the first and final points will be identical.
19    TooFewPointsInRing(RingRole),
20    /// A ring has a self-intersection.
21    SelfIntersection(RingRole),
22    /// One of the Polygon's coordinates is non-finite.
23    NonFiniteCoord(RingRole, CoordIndex),
24    /// A polygon's interiors must be completely within its exterior.
25    InteriorRingNotContainedInExteriorRing(RingRole),
26    /// A valid polygon's rings must not intersect one another. In this case, the intersection is 1-dimensional.
27    IntersectingRingsOnALine(RingRole, RingRole),
28    /// A valid polygon's rings must not intersect one another. In this case, the intersection is 2-dimensional.
29    IntersectingRingsOnAnArea(RingRole, RingRole),
30}
31
32impl fmt::Display for InvalidPolygon {
33    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
34        match self {
35            InvalidPolygon::TooFewPointsInRing(ring) => {
36                write!(f, "{ring} must have at least 3 distinct points")
37            }
38            InvalidPolygon::SelfIntersection(ring) => {
39                write!(f, "{ring} has a self-intersection")
40            }
41            InvalidPolygon::NonFiniteCoord(ring, idx) => {
42                write!(f, "{ring} has a non-finite coordinate at index {}", idx.0)
43            }
44            InvalidPolygon::InteriorRingNotContainedInExteriorRing(ring) => {
45                write!(f, "{ring} is not contained within the polygon's exterior")
46            }
47            InvalidPolygon::IntersectingRingsOnALine(ring_1, ring_2) => {
48                write!(f, "{ring_1} and {ring_2} intersect on a line")
49            }
50            InvalidPolygon::IntersectingRingsOnAnArea(ring_1, ring_2) => {
51                write!(f, "{ring_1} and {ring_2} intersect on an area")
52            }
53        }
54    }
55}

59impl<F: GeoFloat> Validation for Polygon<F> {
60    type Error = InvalidPolygon;
61
62    fn visit_validation<T>(
63        &self,
64        mut handle_validation_error: Box<dyn FnMut(Self::Error) -> Result<(), T> + '_>,
65    ) -> Result<(), T> {
66        if self.is_empty() {
67            return Ok(());
68        }
69
70        for (ring_idx, ring) in std::iter::once(self.exterior())
71            .chain(self.interiors().iter())
72            .enumerate()
73        {
74            if ring.is_empty() {
75                continue;
76            }
77            let ring_role = if ring_idx == 0 {
78                RingRole::Exterior
79            } else {
80                RingRole::Interior(ring_idx - 1)
81            };
82
83            // Perform the various checks
84            if utils::check_too_few_points(ring, true) {
85                handle_validation_error(InvalidPolygon::TooFewPointsInRing(ring_role))?;
86            }
87
88            if utils::linestring_has_self_intersection(ring) {
89                handle_validation_error(InvalidPolygon::SelfIntersection(ring_role))?;
90            }
91
92            for (coord_idx, coord) in ring.0.iter().enumerate() {
93                if utils::check_coord_is_not_finite(coord) {
94                    handle_validation_error(InvalidPolygon::NonFiniteCoord(
95                        ring_role,
96                        CoordIndex(coord_idx),
97                    ))?;
98                }
99            }
100        }
101
102        let polygon_exterior = Polygon::new(self.exterior().clone(), vec![]);
103
104        for (interior_1_idx, interior_1) in self.interiors().iter().enumerate() {
105            let ring_role_1 = RingRole::Interior(interior_1_idx);
106            if interior_1.is_empty() {
107                continue;
108            }
109            let exterior_vs_interior = polygon_exterior.relate(interior_1);
110
111            if !exterior_vs_interior.is_contains() {
112                handle_validation_error(InvalidPolygon::InteriorRingNotContainedInExteriorRing(
113                    ring_role_1,
114                ))?;
115            }
116
117            // Interior ring and exterior ring may only touch at point (not as a line)
118            // and not cross
119            if exterior_vs_interior.get(CoordPos::OnBoundary, CoordPos::Inside)
120                == Dimensions::OneDimensional
121            {
122                handle_validation_error(InvalidPolygon::IntersectingRingsOnALine(
123                    RingRole::Exterior,
124                    ring_role_1,
125                ))?;
126            }
127
128            // PERF: consider using PreparedGeometry
129            let interior_1_as_poly = Polygon::new(interior_1.clone(), vec![]);
130
131            for (interior_2_idx, interior_2) in
132                self.interiors().iter().enumerate().skip(interior_1_idx + 1)
133            {
134                let ring_role_2 = RingRole::Interior(interior_2_idx);
135                let interior_2_as_poly = Polygon::new(interior_2.clone(), vec![]);
136                let intersection_matrix = interior_1_as_poly.relate(&interior_2_as_poly);
137
138                if intersection_matrix.get(CoordPos::Inside, CoordPos::Inside)
139                    == Dimensions::TwoDimensional
140                {
141                    handle_validation_error(InvalidPolygon::IntersectingRingsOnAnArea(
142                        ring_role_1,
143                        ring_role_2,
144                    ))?;
145                }
146                if intersection_matrix.get(CoordPos::OnBoundary, CoordPos::OnBoundary)
147                    == Dimensions::OneDimensional
148                {
149                    handle_validation_error(InvalidPolygon::IntersectingRingsOnALine(
150                        ring_role_1,
151                        ring_role_2,
152                    ))?;
153                }
154            }
155        }
156        Ok(())
157    }
158}
