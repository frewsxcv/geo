14/// Collection of parameters that influence the precision of the algorithm in some sense (see
15/// explanations on fields of this struct)
16///
17/// This implements the `Default` trait and you can just use it most of the time
18#[derive(Debug, Clone)]
19pub struct DelaunayTriangulationConfig<T: SpadeTriangulationFloat> {
20    /// Coordinates within this radius are snapped to the same position. For any two `Coords` there's
21    /// no real way to influence the decision when choosing the snapper and the snappee
22    pub snap_radius: T,
23}
24
25impl<T> Default for DelaunayTriangulationConfig<T>
26where
27    T: SpadeTriangulationFloat,
28{
29    fn default() -> Self {
30        Self {
31            snap_radius: <T as std::convert::From<f32>>::from(0.000_1),
32        }
33    }
34}

69    pub(crate) type CoordsIter<'a, T> = Box<dyn Iterator<Item = Coord<T>> + 'a>;
70
71    pub trait TriangulationRequirementTrait<'a, T>
72    where
73        T: SpadeTriangulationFloat,
74    {
75        /// collect all the lines that are relevant for triangulations from the geometric object that
76        /// should be triangulated.
77        ///
78        /// intersecting lines are allowed
79        fn lines(&'a self) -> Vec<Line<T>>;
80        /// collect all the coords that are relevant for triangulations from the geometric object that
81        /// should be triangulated
82        fn coords(&'a self) -> CoordsIter<'a, T>;
83        /// define a predicate that decides if a point is inside of the object (used for constrained triangulation)
84        fn contains_point(&'a self, p: Point<T>) -> bool;
85
86        // processing of the lines that prepare the lines for triangulation.
87        //
88        // `spade` has the general limitation that constraint lines cannot intersect or else it
89        // will panic. This is why we need to manually split up the lines into smaller parts at the
90        // intersection point
91        //
92        // there's also a preprocessing step which tries to minimize the risk of failure of the algo
93        // through edge cases (thin/flat triangles are prevented as much as possible & lines are deduped, ...)
94        fn cleanup_lines(lines: Vec<Line<T>>, snap_radius: T) -> TriangulationResult<Vec<Line<T>>> {
95            let (known_coords, lines) = preprocess_lines(lines, snap_radius);
96            prepare_intersection_contraint(lines, known_coords, snap_radius)
97        }
98    }
99}

101/// Triangulate polygons using a [Delaunay
102/// Triangulation](https://en.wikipedia.org/wiki/Delaunay_triangulation)
103///
104/// This trait contains both constrained and unconstrained triangulation methods. To read more
105/// about the differences of these methods also consult [this
106/// page](https://en.wikipedia.org/wiki/Constrained_Delaunay_triangulation)
107pub trait TriangulateDelaunay<'a, T>: private::TriangulationRequirementTrait<'a, T>
108where
109    T: SpadeTriangulationFloat,
110{
111    /// returns a triangulation that's solely based on the points of the geometric object
112    ///
113    /// The triangulation is guaranteed to be Delaunay
114    ///
115    /// Note that the lines of the triangulation don't necessarily follow the lines of the input
116    /// geometry. If you wish to achieve that take a look at the `constrained_triangulation` and the
117    /// `constrained_outer_triangulation` functions.
118    ///
119    /// ```rust
120    /// use geo::TriangulateDelaunay;
121    /// use geo::{Polygon, LineString, Coord};
122    /// let u_shape = Polygon::new(
123    ///     LineString::new(vec![
124    ///         Coord { x: 0.0, y: 0.0 },
125    ///         Coord { x: 1.0, y: 0.0 },
126    ///         Coord { x: 1.0, y: 1.0 },
127    ///         Coord { x: 2.0, y: 1.0 },
128    ///         Coord { x: 2.0, y: 0.0 },
129    ///         Coord { x: 3.0, y: 0.0 },
130    ///         Coord { x: 3.0, y: 3.0 },
131    ///         Coord { x: 0.0, y: 3.0 },
132    ///     ]),
133    ///     vec![],
134    /// );
135    /// let unconstrained_triangulation = u_shape.unconstrained_triangulation().unwrap();
136    /// let num_triangles = unconstrained_triangulation.len();
137    /// assert_eq!(num_triangles, 8);
138    /// ```
139    ///
140    fn unconstrained_triangulation(&'a self) -> TriangulationResult<Triangles<T>> {
141        let points = self.coords();
142        points
143            .into_iter()
144            .map(to_spade_point)
145            .try_fold(DelaunayTriangulation::<Point2<T>>::new(), |mut tris, p| {
146                tris.insert(p).map_err(TriangulationError::SpadeError)?;
147                Ok(tris)
148            })
149            .map(triangulation_to_triangles)
150    }
151
152    /// returns triangulation that's based on the points of the geometric object and also
153    /// incorporates the lines of the input geometry
154    ///
155    /// The triangulation is not guaranteed to be Delaunay because of the constraint lines
156    ///
157    /// This outer triangulation also contains triangles that are not included in the input
158    /// geometry if it wasn't convex. Here's an example:
159    ///
160    /// ```text
161    /// ┌──────────────────┐
162    /// │\              __/│
163    /// │ \          __/ / │
164    /// │  \      __/   /  │
165    /// │   \  __/     /   │
166    /// │    \/       /    │
167    /// │     ┌──────┐     │
168    /// │    /│\:::::│\    │
169    /// │   / │:\::::│ \   │
170    /// │  /  │::\:::│  \  │
171    /// │ /   │:::\::│   \ │
172    /// │/    │::::\:│    \│
173    /// └─────┘______└─────┘
174    /// ```
175    ///
176    /// ```rust
177    /// use geo::TriangulateDelaunay;
178    /// use geo::{Polygon, LineString, Coord};
179    /// let u_shape = Polygon::new(
180    ///     LineString::new(vec![
181    ///         Coord { x: 0.0, y: 0.0 },
182    ///         Coord { x: 1.0, y: 0.0 },
183    ///         Coord { x: 1.0, y: 1.0 },
184    ///         Coord { x: 2.0, y: 1.0 },
185    ///         Coord { x: 2.0, y: 0.0 },
186    ///         Coord { x: 3.0, y: 0.0 },
187    ///         Coord { x: 3.0, y: 3.0 },
188    ///         Coord { x: 0.0, y: 3.0 },
189    ///     ]),
190    ///     vec![],
191    /// );
192    /// // we use the default [`SpadeTriangulationConfig`] here
193    /// let constrained_outer_triangulation =
194    /// u_shape.constrained_outer_triangulation(Default::default()).unwrap();
195    /// let num_triangles = constrained_outer_triangulation.len();
196    /// assert_eq!(num_triangles, 8);
197    /// ```
198    ///
199    /// The outer triangulation of the top down U-shape contains extra triangles marked
200    /// with ":". If you want to exclude those, take a look at `constrained_triangulation`
201    fn constrained_outer_triangulation(
202        &'a self,
203        config: DelaunayTriangulationConfig<T>,
204    ) -> TriangulationResult<Triangles<T>> {
205        let lines = self.lines();
206        let lines = Self::cleanup_lines(lines, config.snap_radius)?;
207        lines
208            .into_iter()
209            .map(to_spade_line)
210            .try_fold(
211                ConstrainedDelaunayTriangulation::<Point2<T>>::new(),
212                |mut cdt, [start, end]| {
213                    let start = cdt.insert(start).map_err(TriangulationError::SpadeError)?;
214                    let end = cdt.insert(end).map_err(TriangulationError::SpadeError)?;
215                    // safety check (to prevent panic) whether we can add the line
216                    if !cdt.can_add_constraint(start, end) {
217                        return Err(TriangulationError::ConstraintFailure);
218                    }
219                    cdt.add_constraint(start, end);
220                    Ok(cdt)
221                },
222            )
223            .map(triangulation_to_triangles)
224    }
225
226    /// returns triangulation that's based on the points of the geometric object and also
227    /// incorporates the lines of the input geometry
228    ///
229    /// The triangulation is not guaranteed to be Delaunay because of the constraint lines
230    ///
231    /// This triangulation only contains triangles that are included in the input geometry.
232    /// Here's an example:
233    ///
234    /// ```text
235    /// ┌──────────────────┐
236    /// │\              __/│
237    /// │ \          __/ / │
238    /// │  \      __/   /  │
239    /// │   \  __/     /   │
240    /// │    \/       /    │
241    /// │     ┌──────┐     │
242    /// │    /│      │\    │
243    /// │   / │      │ \   │
244    /// │  /  │      │  \  │
245    /// │ /   │      │   \ │
246    /// │/    │      │    \│
247    /// └─────┘      └─────┘
248    /// ```
249    ///
250    /// ```rust
251    /// use geo::TriangulateDelaunay;
252    /// use geo::{Polygon, LineString, Coord};
253    /// let u_shape = Polygon::new(
254    ///     LineString::new(vec![
255    ///         Coord { x: 0.0, y: 0.0 },
256    ///         Coord { x: 1.0, y: 0.0 },
257    ///         Coord { x: 1.0, y: 1.0 },
258    ///         Coord { x: 2.0, y: 1.0 },
259    ///         Coord { x: 2.0, y: 0.0 },
260    ///         Coord { x: 3.0, y: 0.0 },
261    ///         Coord { x: 3.0, y: 3.0 },
262    ///         Coord { x: 0.0, y: 3.0 },
263    ///     ]),
264    ///     vec![],
265    /// );
266    /// // we use the default [`DelaunayTriangulationConfig`] here
267    /// let constrained_triangulation = u_shape.constrained_triangulation(Default::default()).unwrap();
268    /// let num_triangles = constrained_triangulation.len();
269    /// assert_eq!(num_triangles, 6);
270    /// ```
271    ///
272    /// Compared to the `constrained_outer_triangulation` it only includes the triangles
273    /// inside of the input geometry
274    fn constrained_triangulation(
275        &'a self,
276        config: DelaunayTriangulationConfig<T>,
277    ) -> TriangulationResult<Triangles<T>> {
278        self.constrained_outer_triangulation(config)
279            .map(|triangles| {
280                triangles
281                    .into_iter()
282                    .filter(|triangle| {
283                        let center = triangle.centroid();
284                        self.contains_point(center)
285                    })
286                    .collect::<Vec<_>>()
287            })
288    }
289}

291/// conversion from spade triangulation back to geo triangles
292fn triangulation_to_triangles<T, F>(triangulation: T) -> Triangles<F>
293where
294    T: Triangulation<Vertex = Point2<F>>,
295    F: SpadeTriangulationFloat,
296{
297    triangulation
298        .inner_faces()
299        .map(|face| face.positions())
300        .map(|points| points.map(|p| Coord::<F> { x: p.x, y: p.y }))
301        .map(Triangle::from)
302        .collect::<Vec<_>>()
303}

375fn prepare_intersection_contraint<T: SpadeTriangulationFloat>(
376    mut lines: Vec<Line<T>>,
377    mut known_points: Vec<Coord<T>>,
378    snap_radius: T,
379) -> Result<Vec<Line<T>>, TriangulationError> {
380    // Rule 2 of "Power of 10" rules (NASA)
381    // safety net. We can't prove that the `while let` loop isn't going to run infinitely, so
382    // we abort after a fixed amount of iterations. In case that the iteration seems to loop
383    // indefinitely this check will return an Error indicating the infinite loop.
384    let mut loop_count = 1000;
385    let mut loop_check = || {
386        loop_count -= 1;
387        (loop_count != 0)
388            .then_some(())
389            .ok_or(TriangulationError::LoopTrap)
390    };
391
392    while let Some((indices, intersection)) = {
393        let mut iter = iter_line_pairs(&lines);
394        iter.find_map(find_intersecting_lines_fn)
395    } {
396        loop_check()?;
397        let [l0, l1] = remove_lines_by_index(indices, &mut lines);
398        let new_lines = split_lines([l0, l1], intersection);
399        let new_lines = cleanup_filter_lines(new_lines, &lines, &mut known_points, snap_radius);
400
401        lines.extend(new_lines);
402    }
403
404    Ok(lines)
405}

407/// iterates over all combinations (a,b) of lines in a vector where a != b
408fn iter_line_pairs<T: SpadeTriangulationFloat>(
409    lines: &[Line<T>],
410) -> impl Iterator<Item = [(usize, &Line<T>); 2]> {
411    lines.iter().enumerate().flat_map(|(idx0, line0)| {
412        lines
413            .iter()
414            .enumerate()
415            .skip(idx0 + 1)
416            .filter(move |(idx1, line1)| *idx1 != idx0 && line0 != *line1)
417            .map(move |(idx1, line1)| [(idx0, line0), (idx1, line1)])
418    })
419}

421/// checks whether two lines are intersecting and if so, checks the intersection to not be ill
422/// formed
423///
424/// returns
425/// - [usize;2] : sorted indexes of lines, smaller one comes first
426/// - intersection : type of intersection
427fn find_intersecting_lines_fn<T: SpadeTriangulationFloat>(
428    [(idx0, line0), (idx1, line1)]: [(usize, &Line<T>); 2],
429) -> Option<([usize; 2], LineIntersection<T>)> {
430    line_intersection(*line0, *line1)
431        .filter(|intersection| {
432            match intersection {
433                // intersection is not located in both lines
434                LineIntersection::SinglePoint { is_proper, .. } if !is_proper => false,
435                // collinear intersection is length zero line
436                LineIntersection::Collinear { intersection }
437                    if intersection.start == intersection.end =>
438                {
439                    false
440                }
441                _ => true,
442            }
443        })
444        .map(|intersection| ([idx0, idx1], intersection))
445}

447/// removes two lines by index in a safe way since the second index can be invalidated after
448/// the first line was removed (remember `.remove(idx)` returns the element and shifts the tail
449/// of the vector in direction of its start to close the gap)
450fn remove_lines_by_index<T: SpadeTriangulationFloat>(
451    mut indices: [usize; 2],
452    lines: &mut Vec<Line<T>>,
453) -> [Line<T>; 2] {
454    indices.sort();
455    let [idx0, idx1] = indices;
456    let l1 = lines.remove(idx1);
457    let l0 = lines.remove(idx0);
458    [l0, l1]
459}

461/// split lines based on intersection kind:
462///
463/// - intersection point: create 4 new lines from the existing line's endpoints to the intersection
464///   point
465/// - collinear: create 3 new lines (before overlap, overlap, after overlap)
466fn split_lines<T: SpadeTriangulationFloat>(
467    [l0, l1]: [Line<T>; 2],
468    intersection: LineIntersection<T>,
469) -> Vec<Line<T>> {
470    match intersection {
471        LineIntersection::SinglePoint { intersection, .. } => [
472            (l0.start, intersection),
473            (l0.end, intersection),
474            (l1.start, intersection),
475            (l1.end, intersection),
476        ]
477        .map(|(a, b)| Line::new(a, b))
478        .to_vec(),
479        LineIntersection::Collinear { .. } => {
480            let mut points = [l0.start, l0.end, l1.start, l1.end];
481            // sort points by their coordinate values to resolve ambiguities
482            points.sort_by(|a, b| {
483                a.x.partial_cmp(&b.x)
484                    .expect("sorting points by coordinate x failed")
485                    .then_with(|| {
486                        a.y.partial_cmp(&b.y)
487                            .expect("sorting points by coordinate y failed")
488                    })
489            });
490            // since all points are on one line we can just create new lines from consecutive
491            // points after sorting
492            points
493                .windows(2)
494                .map(|win| Line::new(win[0], win[1]))
495                .collect::<Vec<_>>()
496        }
497    }
498}

500/// new lines from the `split_lines` function may contain a variety of ill formed lines, this
501/// function cleans all of these cases up
502fn cleanup_filter_lines<T: SpadeTriangulationFloat>(
503    lines_need_check: Vec<Line<T>>,
504    existing_lines: &[Line<T>],
505    known_points: &mut Vec<Coord<T>>,
506    snap_radius: T,
507) -> Vec<Line<T>> {
508    lines_need_check
509        .into_iter()
510        .map(|mut line| {
511            line.start = snap_or_register_point(line.start, known_points, snap_radius);
512            line.end = snap_or_register_point(line.end, known_points, snap_radius);
513            line
514        })
515        .filter(|l| l.start != l.end)
516        .filter(|l| !existing_lines.contains(l))
517        .filter(|l| !existing_lines.contains(&Line::new(l.end, l.start)))
518        .collect::<Vec<_>>()
519}

521/// snap point to the nearest existing point if it's close enough
522///
523/// snap_radius can be configured via the third parameter of this function
524fn snap_or_register_point<T: SpadeTriangulationFloat>(
525    point: Coord<T>,
526    known_points: &mut Vec<Coord<T>>,
527    snap_radius: T,
528) -> Coord<T> {
529    known_points
530        .iter()
531        // find closest
532        .min_by(|a, b| {
533            Euclidean
534                .distance(**a, point)
535                .partial_cmp(&Euclidean.distance(**b, point))
536                .expect("Couldn't compare coordinate distances")
537        })
538        // only snap if closest is within epsilon range
539        .filter(|nearest_point| Euclidean.distance(**nearest_point, point) < snap_radius)
540        .cloned()
541        // otherwise register and use input point
542        .unwrap_or_else(|| {
543            known_points.push(point);
544            point
545        })
546}

548/// preprocesses lines so that we're less likely to hit issues when using the spade triangulation
549fn preprocess_lines<T: SpadeTriangulationFloat>(
550    lines: Vec<Line<T>>,
551    snap_radius: T,
552) -> (Vec<Coord<T>>, Vec<Line<T>>) {
553    let mut known_coords: Vec<Coord<T>> = vec![];
554    let capacity = lines.len();
555    let lines = lines
556        .into_iter()
557        .fold(Vec::with_capacity(capacity), |mut lines, mut line| {
558            // deduplicating:
559
560            // 1. snap coords of lines to existing coords
561            line.start = snap_or_register_point(line.start, &mut known_coords, snap_radius);
562            line.end = snap_or_register_point(line.end, &mut known_coords, snap_radius);
563            if
564            // 2. make sure line isn't degenerate (no length when start == end)
565            line.start != line.end
566                // 3. make sure line or flipped line wasn't already added
567                && !lines.contains(&line)
568                && !lines.contains(&Line::new(line.end, line.start))
569            {
570                lines.push(line)
571            }
572
573            lines
574        });
575    (known_coords, lines)
576}

765        let u_shape = Polygon::new(
766            LineString::new(vec![
767                Coord { x: 0.0, y: 0.0 },
768                Coord { x: 1.0, y: 0.0 },
769                Coord { x: 1.0, y: 1.0 },
770                Coord { x: 2.0, y: 1.0 },
771                Coord { x: 2.0, y: 0.0 },
772                Coord { x: 3.0, y: 0.0 },
773                Coord { x: 3.0, y: 3.0 },
774                Coord { x: 0.0, y: 3.0 },
775            ]),
776            vec![],
777        );
778
779        for snap_with in (1..6).map(|pow| 0.1_f64.powi(pow)) {
780            let constrained_triangulation =
781                u_shape.constrained_triangulation(DelaunayTriangulationConfig {
782                    snap_radius: snap_with,
783                });
784            assert_num_triangles(&constrained_triangulation, 6);
785        }
