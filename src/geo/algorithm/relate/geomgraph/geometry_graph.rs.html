16/// The computation of the [`IntersectionMatrix`](crate::algorithm::relate::IntersectionMatrix) relies on the use of a
17/// structure called a "topology graph". The topology graph contains nodes (CoordNode) and
18/// edges (Edge) corresponding to the nodes and line segments of a [`Geometry`](crate::Geometry). Each
19/// node and edge in the graph is labeled with its topological location
20/// relative to the source geometry.
21///
22/// Note that there is no requirement that points of self-intersection be a
23/// vertex. Thus, to obtain a correct topology graph, [`Geometry`](crate::Geometry) must be
24/// self-noded before constructing their graphs.
25///
26/// Two fundamental operations are supported by topology graphs:
27///   - Computing the intersections between all the edges and nodes of a single graph
28///   - Computing the intersections between the edges and nodes of two different graphs
29///
30/// GeometryGraph is based on [JTS's `GeomGraph` as of 1.18.1](https://github.com/locationtech/jts/blob/jts-1.18.1/modules/core/src/main/java/org/locationtech/jts/geomgraph/GeometryGraph.java)
31#[derive(Clone)]
32pub struct GeometryGraph<'a, F>
33where
34    F: GeoFloat,
35{
36    arg_index: usize,
37    parent_geometry: GeometryCow<'a, F>,
38    tree: Option<Rc<RTree<Segment<F>>>>,
39    use_boundary_determination_rule: bool,
40    has_computed_self_nodes: bool,
41    planar_graph: PlanarGraph<F>,
42}

59            .unwrap_or_else(|| Rc::new(self.build_tree()))

62    pub(crate) fn build_tree(&self) -> RTree<Segment<F>> {
63        let segments: Vec<Segment<F>> = self
64            .edges()
65            .iter()
66            .enumerate()
67            .flat_map(|(edge_idx, edge)| {
68                let edge = RefCell::borrow(edge);
69                let start_of_final_segment: usize = edge.coords().len() - 1;
70                (0..start_of_final_segment).map(move |segment_idx| {
71                    let p1 = edge.coords()[segment_idx];
72                    let p2 = edge.coords()[segment_idx + 1];
73                    Segment::new(edge_idx, segment_idx, p1, p2)
74                })
75            })
76            .collect();
77        RTree::bulk_load(segments)
78    }

150    /// Determine whether a component (node or edge) that appears multiple times in elements
151    /// of a Multi-Geometry is in the boundary or the interior of the Geometry
152    pub fn determine_boundary(boundary_count: usize) -> CoordPos {
153        // For now, we only support the SFS "Mod-2 Rule"
154        // We could make this configurable if we wanted to support alternative boundary rules.
155        if boundary_count % 2 == 1 {
156            CoordPos::OnBoundary
157        } else {
158            CoordPos::Inside
159        }
160    }

166    pub(crate) fn add_geometry(&mut self, geometry: &GeometryCow<F>) {
167        if geometry.is_empty() {
168            return;
169        }
170        match geometry {
171            GeometryCow::Line(line) => self.add_line(line),
172            GeometryCow::Rect(rect) => {
173                // PERF: avoid this conversion/clone?
174                self.add_polygon(&rect.to_polygon());
175            }
176            GeometryCow::Point(point) => {
177                self.add_point(point);
178            }
179            GeometryCow::Polygon(polygon) => self.add_polygon(polygon),
180            GeometryCow::Triangle(triangle) => {
181                // PERF: avoid this conversion/clone?
182                self.add_polygon(&triangle.to_polygon());
183            }
184            GeometryCow::LineString(line_string) => self.add_line_string(line_string),
185            GeometryCow::MultiPoint(multi_point) => {
186                for point in &multi_point.0 {
187                    self.add_point(point);
188                }
189            }
190            GeometryCow::MultiPolygon(multi_polygon) => {
191                // check if this Geometry should obey the Boundary Determination Rule
192                // all collections except MultiPolygons obey the rule
193                self.use_boundary_determination_rule = false;
194                for polygon in &multi_polygon.0 {
195                    self.add_polygon(polygon);
196                }
197            }
198            GeometryCow::MultiLineString(multi_line_string) => {
199                for line_string in &multi_line_string.0 {
200                    self.add_line_string(line_string);
201                }
202            }
203            GeometryCow::GeometryCollection(geometry_collection) => {
204                for geometry in geometry_collection.iter() {
205                    self.add_geometry(&GeometryCow::from(geometry));
206                }
207            }
208        }
209    }

211    fn add_polygon_ring(
212        &mut self,
213        linear_ring: &LineString<F>,
214        cw_left: CoordPos,
215        cw_right: CoordPos,
216    ) {
217        debug_assert!(linear_ring.is_closed());
218        if linear_ring.is_empty() {
219            return;
220        }
221
222        let mut coords: Vec<Coord<F>> = Vec::with_capacity(linear_ring.0.len());
223        // remove repeated coords
224        for coord in &linear_ring.0 {
225            if coords.last() != Some(coord) {
226                coords.push(*coord)
227            }
228        }
229
230        if coords.len() < 4 {
231            // TODO: we could return an Err here, but this has ramifications for how we can
232            // use this code in other operations - do we want all our methods, like `contains` to
233            // return a Result?
234            warn!("encountered invalid ring, which has undefined results");
235        }
236        let first_point = coords[0];
237
238        use crate::winding_order::{Winding, WindingOrder};
239        let (left, right) = match linear_ring.winding_order() {
240            Some(WindingOrder::Clockwise) => (cw_left, cw_right),
241            Some(WindingOrder::CounterClockwise) => (cw_right, cw_left),
242            None => {
243                warn!("polygon had no winding order. Results are undefined.");
244                (cw_left, cw_right)
245            }
246        };
247
248        let edge = Edge::new(
249            coords,
250            Label::new(
251                self.arg_index,
252                TopologyPosition::area(CoordPos::OnBoundary, left, right),
253            ),
254        );
255        self.insert_edge(edge);
256
257        // insert the endpoint as a node, to mark that it is on the boundary
258        self.insert_point(self.arg_index, first_point, CoordPos::OnBoundary);
259    }
260
261    fn add_polygon(&mut self, polygon: &Polygon<F>) {
262        self.add_polygon_ring(polygon.exterior(), CoordPos::Outside, CoordPos::Inside);
263        // Holes are topologically labeled opposite to the shell, since
264        // the interior of the polygon lies on their opposite side
265        // (on the left, if the hole is oriented CW)
266        for hole in polygon.interiors() {
267            self.add_polygon_ring(hole, CoordPos::Inside, CoordPos::Outside)
268        }
269    }
270
271    fn add_line_string(&mut self, line_string: &LineString<F>) {
272        if line_string.is_empty() {
273            return;
274        }
275
276        let mut coords: Vec<Coord<F>> = Vec::with_capacity(line_string.0.len());
277        for coord in &line_string.0 {
278            if coords.last() != Some(coord) {
279                coords.push(*coord)
280            }
281        }
282
283        if coords.len() < 2 {
284            warn!("Treating invalid linestring as point, which has undefined results");
285            self.add_point(&coords[0].into());
286            return;
287        }
288
289        self.insert_boundary_point(*coords.first().unwrap());
290        self.insert_boundary_point(*coords.last().unwrap());
291
292        let edge = Edge::new(
293            coords,
294            Label::new(
295                self.arg_index,
296                TopologyPosition::line_or_point(CoordPos::Inside),
297            ),
298        );
299        self.insert_edge(edge);
300    }
301
302    fn add_line(&mut self, line: &Line<F>) {
303        self.insert_boundary_point(line.start);
304        self.insert_boundary_point(line.end);
305
306        let edge = Edge::new(
307            vec![line.start, line.end],
308            Label::new(
309                self.arg_index,
310                TopologyPosition::line_or_point(CoordPos::Inside),
311            ),
312        );
313
314        self.insert_edge(edge);
315    }

319    fn add_point(&mut self, point: &Point<F>) {
320        self.insert_point(self.arg_index, (*point).into(), CoordPos::Inside);

323    /// Compute self-nodes, taking advantage of the Geometry type to minimize the number of
324    /// intersection tests.  (E.g. rings are not tested for self-intersection, since they are
325    /// assumed to be valid).
326    ///
327    /// `line_intersector` the [`LineIntersector`] to use to determine intersection
328    pub(crate) fn compute_self_nodes(&mut self, line_intersector: Box<dyn LineIntersector<F>>) {
329        if self.has_computed_self_nodes {
330            return;
331        }
332        self.has_computed_self_nodes = true;
333
334        let mut segment_intersector = SegmentIntersector::new(line_intersector, true);
335
336        // optimize intersection search for valid Polygons and LinearRings
337        let is_rings = match self.geometry() {
338            GeometryCow::LineString(ls) => ls.is_closed(),
339            GeometryCow::MultiLineString(ls) => ls.is_closed(),
340            GeometryCow::Polygon(_) | GeometryCow::MultiPolygon(_) => true,
341            _ => false,
342        };
343        let check_for_self_intersecting_edges = !is_rings;
344
345        let edge_set_intersector = RStarEdgeSetIntersector;
346        edge_set_intersector.compute_intersections_within_set(
347            self,
348            check_for_self_intersecting_edges,
349            &mut segment_intersector,
350        );
351        self.add_self_intersection_nodes();
352    }

354    pub(crate) fn compute_edge_intersections(
355        &self,
356        other: &GeometryGraph<F>,
357        line_intersector: Box<dyn LineIntersector<F>>,
358    ) -> SegmentIntersector<F> {
359        let mut segment_intersector = SegmentIntersector::new(line_intersector, false);
360        segment_intersector.set_boundary_nodes(
361            self.boundary_nodes().cloned().collect(),
362            other.boundary_nodes().cloned().collect(),
363        );
364
365        let edge_set_intersector = RStarEdgeSetIntersector;
366        edge_set_intersector.compute_intersections_between_sets(
367            self,
368            other,
369            &mut segment_intersector,
370        );
371
372        segment_intersector
373    }

375    fn insert_point(&mut self, arg_index: usize, coord: Coord<F>, position: CoordPos) {
376        let node: &mut CoordNode<F> = self.add_node_with_coordinate(coord);
377        node.label_mut().set_on_position(arg_index, position);

380    /// Add the boundary points of 1-dim (line) geometries.
381    fn insert_boundary_point(&mut self, coord: Coord<F>) {
382        let arg_index = self.arg_index;
383        let node: &mut CoordNode<F> = self.add_node_with_coordinate(coord);
384
385        let label: &mut Label = node.label_mut();
386
387        // determine the current location for the point (if any)
388        let boundary_count = {
389            #[allow(clippy::bool_to_int_with_if)]
390            let prev_boundary_count =
391                if Some(CoordPos::OnBoundary) == label.position(arg_index, Direction::On) {
392                    1
393                } else {
394                    0
395                };
396            prev_boundary_count + 1
397        };
398
399        let new_position = Self::determine_boundary(boundary_count);
400        label.set_on_position(arg_index, new_position);
401    }

403    fn add_self_intersection_nodes(&mut self) {
404        let positions_and_intersections: Vec<(CoordPos, Vec<Coord<F>>)> = self
405            .edges()
406            .iter()
407            .map(|cell| cell.borrow())
408            .map(|edge| {
409                let position = edge
410                    .label()
411                    .on_position(self.arg_index)
412                    .expect("all edge labels should have an `on` position by now");
413                let coordinates = edge
414                    .edge_intersections()
415                    .iter()
416                    .map(|edge_intersection| edge_intersection.coordinate());
417
418                (position, coordinates.collect())
419            })
420            .collect();
421
422        for (position, edge_intersection_coordinates) in positions_and_intersections {
423            for coordinate in edge_intersection_coordinates {
424                self.add_self_intersection_node(coordinate, position)
425            }
426        }
427    }

429    /// Add a node for a self-intersection.
430    ///
431    /// If the node is a potential boundary node (e.g. came from an edge which is a boundary), then
432    /// insert it as a potential boundary node.  Otherwise, just add it as a regular node.
433    fn add_self_intersection_node(&mut self, coord: Coord<F>, position: CoordPos) {
434        // if this node is already a boundary node, don't change it
435        if self.is_boundary_node(coord) {
436            return;
437        }
438
439        if position == CoordPos::OnBoundary && self.use_boundary_determination_rule {
440            self.insert_boundary_point(coord)
441        } else {
442            self.insert_point(self.arg_index, coord, position)
443        }
444    }
