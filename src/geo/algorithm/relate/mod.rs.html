14/// Topologically relate two geometries based on [DE-9IM](https://en.wikipedia.org/wiki/DE-9IM) semantics.
15///
16/// See [`IntersectionMatrix`] for details. All predicates are available on the calculated matrix.
17///
18/// # Examples
19///
20/// ```
21/// use geo::{coord, Line, Rect, line_string};
22/// use crate::geo::relate::Relate;
23///
24/// let line = Line::new(coord! { x: 2.0, y: 2.0}, coord! { x: 4.0, y: 4.0 });
25/// let rect = Rect::new(coord! { x: 2.0, y: 2.0}, coord! { x: 4.0, y: 4.0 });
26/// let intersection_matrix = rect.relate(&line);
27///
28/// assert!(intersection_matrix.is_intersects());
29/// assert!(!intersection_matrix.is_disjoint());
30/// assert!(intersection_matrix.is_contains());
31/// assert!(!intersection_matrix.is_within());
32///
33/// let line = Line::new(coord! { x: 1.0, y: 1.0}, coord! { x: 5.0, y: 5.0 });
34/// let rect = Rect::new(coord! { x: 2.0, y: 2.0}, coord! { x: 4.0, y: 4.0 });
35/// let intersection_matrix = rect.relate(&line);
36/// assert!(intersection_matrix.is_intersects());
37/// assert!(!intersection_matrix.is_disjoint());
38/// assert!(!intersection_matrix.is_contains());
39/// assert!(!intersection_matrix.is_within());
40///
41/// let rect_boundary = line_string![
42///     (x: 2.0, y: 2.0),
43///     (x: 4.0, y: 2.0),
44///     (x: 4.0, y: 4.0),
45///     (x: 2.0, y: 4.0),
46///     (x: 2.0, y: 2.0)
47/// ];
48/// let intersection_matrix = rect.relate(&rect_boundary);
49/// assert!(intersection_matrix.is_intersects());
50/// assert!(!intersection_matrix.is_disjoint());
51/// // According to DE-9IM, polygons don't contain their own boundary
52/// assert!(!intersection_matrix.is_contains());
53/// assert!(!intersection_matrix.is_within());
54/// ```
55///
56/// Note: `Relate` must not be called on geometries containing `NaN` coordinates.
57pub trait Relate<F: GeoFloat>: BoundingRect<F> + HasDimensions {
58    /// Returns a noded topology graph for the geometry.
59    ///
60    /// # Params
61    ///
62    /// `idx`: 0 or 1, designating A or B (respectively) in the role this geometry plays
63    ///        in the relation. e.g. in `a.relate(b)`
64    fn geometry_graph(&self, idx: usize) -> GeometryGraph<F>;
65
66    fn relate(&self, other: &impl Relate<F>) -> IntersectionMatrix
67    where
68        Self: Sized,
69    {
70        RelateOperation::new(self, other).compute_intersection_matrix()
71    }
72}
73
74macro_rules! relate_impl {
75    ($($t:ty ,)*) => {
76        $(
77            impl<F: GeoFloat> Relate<F> for $t {
78                fn geometry_graph(&self, arg_index: usize) -> GeometryGraph<F> {
79                    $crate::relate::GeometryGraph::new(arg_index, GeometryCow::from(self))
80                }
81            }
82            impl<F: GeoFloat> From<$t> for PreparedGeometry<'static, $t, F> {
83                fn from(geometry: $t) -> Self {
84                    $crate::relate::geomgraph::index::prepare_geometry(geometry)
85                }
86            }
87            impl<'a, F: GeoFloat> From<&'a $t> for PreparedGeometry<'a, &'a $t, F> {
88                fn from(geometry: &'a $t) -> Self {
89                    $crate::relate::geomgraph::index::prepare_geometry(geometry)
90                }
91            }
92        )*
93    };
94}
