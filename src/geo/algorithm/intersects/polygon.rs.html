20impl<T> Intersects<Line<T>> for Polygon<T>
21where
22    T: GeoNum,
23{
24    fn intersects(&self, line: &Line<T>) -> bool {
25        self.exterior().intersects(line)
26            || self.interiors().iter().any(|inner| inner.intersects(line))
27            || self.intersects(&line.start)
28            || self.intersects(&line.end)
29    }
30}

55impl<T> Intersects<Polygon<T>> for Polygon<T>
56where
57    T: GeoNum,
58{
59    fn intersects(&self, polygon: &Polygon<T>) -> bool {
60        if has_disjoint_bboxes(self, polygon) {
61            return false;
62        }
63
64        // self intersects (or contains) any line in polygon
65        self.intersects(polygon.exterior()) ||
66            polygon.interiors().iter().any(|inner_line_string| self.intersects(inner_line_string)) ||
67            // self is contained inside polygon
68            polygon.intersects(self.exterior())
69    }
70}

74impl<G, T> Intersects<G> for MultiPolygon<T>
75where
76    T: GeoNum,
77    Polygon<T>: Intersects<G>,
78    G: BoundingRect<T>,
79{
80    fn intersects(&self, rhs: &G) -> bool {
81        if has_disjoint_bboxes(self, rhs) {
82            return false;
83        }
84        self.iter().any(|p| p.intersects(rhs))
85    }
86}
