44/// Calculate the Euclidean distance (a.k.a. pythagorean distance) between two Points
45impl<F: CoordFloat> Distance<F, Point<F>, Point<F>> for Euclidean {
46    /// Calculate the Euclidean distance (a.k.a. pythagorean distance) between two Points
47    ///
48    /// # Units
49    /// - `origin`, `destination`: Point where the units of x/y represent non-angular units
50    ///    — e.g. meters or miles, not lon/lat. For lon/lat points, use the
51    ///    [`Haversine`] or [`Geodesic`] [metric spaces].
52    /// - returns: distance in the same units as the `origin` and `destination` points
53    ///
54    /// # Example
55    /// ```
56    /// use geo::{Euclidean, Distance};
57    /// use geo::Point;
58    /// // web mercator
59    /// let new_york_city = Point::new(-8238310.24, 4942194.78);
60    /// // web mercator
61    /// let london = Point::new(-14226.63, 6678077.70);
62    /// let distance: f64 = Euclidean.distance(new_york_city, london);
63    ///
64    /// assert_eq!(
65    ///     8_405_286., // meters in web mercator
66    ///     distance.round()
67    /// );
68    /// ```
69    ///
70    /// [`Haversine`]: crate::line_measures::metric_spaces::Haversine
71    /// [`Geodesic`]: crate::line_measures::metric_spaces::Geodesic
72    /// [metric spaces]: crate::line_measures::metric_spaces
73    fn distance(&self, origin: Point<F>, destination: Point<F>) -> F {
74        self.distance(origin.0, destination.0)
75    }
76}

96impl<F: GeoFloat> Distance<F, &Point<F>, &Polygon<F>> for Euclidean {
97    fn distance(&self, point: &Point<F>, polygon: &Polygon<F>) -> F {
98        // No need to continue if the polygon intersects the point, or is zero-length
99        if polygon.exterior().0.is_empty() || polygon.intersects(point) {
100            return F::zero();
101        }
102        // fold the minimum interior ring distance if any, followed by the exterior
103        // shell distance, returning the minimum of the two distances
104        polygon
105            .interiors()
106            .iter()
107            .map(|ring| self.distance(point, ring))
108            .fold(Bounded::max_value(), |accum: F, val| accum.min(val))
109            .min(
110                polygon
111                    .exterior()
112                    .lines()
113                    .map(|line| {
114                        ::geo_types::private_utils::line_segment_distance(
115                            point.0, line.start, line.end,
116                        )
117                    })
118                    .fold(Bounded::max_value(), |accum, val| accum.min(val)),
119            )
120    }
121}

130impl<F: GeoFloat> Distance<F, &Line<F>, &Line<F>> for Euclidean {
131    fn distance(&self, line_a: &Line<F>, line_b: &Line<F>) -> F {
132        if line_a.intersects(line_b) {
133            return F::zero();
134        }
135        // minimum of all Point-Line distances
136        self.distance(&line_a.start_point(), line_b)
137            .min(self.distance(&line_a.end_point(), line_b))
138            .min(self.distance(&line_b.start_point(), line_a))
139            .min(self.distance(&line_b.end_point(), line_a))
140    }
141}

185impl<F: GeoFloat> Distance<F, &LineString<F>, &Polygon<F>> for Euclidean {
186    fn distance(&self, line_string: &LineString<F>, polygon: &Polygon<F>) -> F {
187        if line_string.intersects(polygon) {
188            F::zero()
189        } else if !polygon.interiors().is_empty()
190            // FIXME: Explodes on empty line_string
191            && ring_contains_coord(polygon.exterior(), line_string.0[0])
192        {
193            // check each ring distance, returning the minimum
194            let mut mindist: F = Float::max_value();
195            for ring in polygon.interiors() {
196                mindist = mindist.min(nearest_neighbour_distance(line_string, ring))
197            }
198            mindist
199        } else {
200            nearest_neighbour_distance(line_string, polygon.exterior())
201        }
202    }
203}

362/// Uses an R* tree and nearest-neighbour lookups to calculate minimum distances
363// This is somewhat slow and memory-inefficient, but certainly better than quadratic time
364fn nearest_neighbour_distance<F: GeoFloat>(geom1: &LineString<F>, geom2: &LineString<F>) -> F {
365    let tree_a = RTree::bulk_load(geom1.lines().map(CachedEnvelope::new).collect());
366    let tree_b = RTree::bulk_load(geom2.lines().map(CachedEnvelope::new).collect());
367    // Return minimum distance between all geom a points and geom b lines, and all geom b points and geom a lines
368    geom2
369        .points()
370        .fold(Bounded::max_value(), |acc: F, point| {
371            let nearest = tree_a.nearest_neighbor(&point).unwrap();
372            acc.min(Euclidean.distance(nearest as &Line<F>, &point))
373        })
374        .min(geom1.points().fold(Bounded::max_value(), |acc, point| {
375            let nearest = tree_b.nearest_neighbor(&point).unwrap();
376            acc.min(Euclidean.distance(nearest as &Line<F>, &point))
377        }))
378}

380fn ring_contains_coord<T: GeoNum>(ring: &LineString<T>, c: Coord<T>) -> bool {
381    match coord_pos_relative_to_ring(c, ring) {
382        CoordPos::Inside => true,
383        CoordPos::OnBoundary | CoordPos::Outside => false,
384    }
