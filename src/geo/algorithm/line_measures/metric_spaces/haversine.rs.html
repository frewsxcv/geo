7/// Use the [`Haversine`] constant (an instance of `HaversineMeasure`) rather than building your own
8/// customized [`HaversineMeasure`] for standard spherical Earth measurements.
9///
10/// `HaversineMeasure` measures distance on a sphere using the [haversine formula].
11/// Distances are considered [great circle] lengths and given in units that match those of
12/// the `radius` passed to [`HaversineMeasure::new`] (typically meters).
13///
14/// You may specify a custom radius for the Earth (or other sphere), but for normal spherical
15/// measurements of the Earth, you should use the simpler [`Haversine`] which uses the standard
16/// earth radius of 6371.0088 km (6_371_008.7714 m), based on the recommendation of the IUGG.
17///
18/// # Examples
19///
20/// ```
21/// # use approx::assert_relative_eq;
22/// use geo::{wkt, HaversineMeasure, Haversine, Distance};
23///
24/// let start = wkt!(POINT(23.319941 42.698334)); // Sofia: Longitude, Latitude
25/// let finish = wkt!(POINT(24.742168 42.136097)); // Plovdiv: Longitude, Latitude
26///
27/// // Typically, you can use `Haversine` for measuring on the Earth's surface.
28/// assert_relative_eq!(
29///     132433.09929460194,
30///     Haversine.distance(start, finish)
31/// );
32///
33/// // The default `Haversine` const has a radius equal to the mean radius of the GRS80 ellipsoid (6371.0088 km).
34/// assert_relative_eq!(
35///     Haversine.radius(),
36///     HaversineMeasure::GRS80_MEAN_RADIUS.radius()
37/// );
38///
39/// // You may choose to use one of the other well known estimations of the Earth's radius,
40/// // which may result in *slightly* different results.
41/// assert_relative_eq!(
42///     132433.06564071847,
43///     HaversineMeasure::GRS80_EQUAL_AREA.distance(start, finish)
44/// );
45///
46/// // Or you can specify whatever radius you want to get some "out of this world" results.
47/// let mars_sphere = HaversineMeasure::new(3_389_500.0); // 👽 Mars radius in meters
48/// assert_relative_eq!(
49///     70456.97222377927,
50///     mars_sphere.distance(start, finish)
51/// );
52/// ```
53///
54/// # References
55///
56/// Moritz, H. (2000). Geodetic Reference System 1980. Journal of Geodesy, 74(1), 128–133. doi:10.1007/s001900050278
57/// "Derived Geometric Constants: **R1: mean radius**" (p131)
58/// - <https://link.springer.com/article/10.1007%2Fs001900050278>
59/// - <https://sci-hub.se/https://doi.org/10.1007/s001900050278>
60///
61/// [haversine formula]: https://en.wikipedia.org/wiki/Haversine_formula
62/// [great circle]: https://en.wikipedia.org/wiki/Great_circle
63pub struct HaversineMeasure {
64    radius: f64,
65}
66
67impl Default for HaversineMeasure {
68    fn default() -> Self {
69        HaversineMeasure::GRS80_MEAN_RADIUS
70    }
71}

157impl<F: CoordFloat + FromPrimitive> Bearing<F> for HaversineMeasure {
158    /// Returns the bearing from `origin` to `destination` in degrees along a [great circle].
159    ///
160    /// # Units
161    ///
162    /// - `origin`, `destination`: Points where x/y are lon/lat degree coordinates
163    /// - returns: degrees, where: North: 0°, East: 90°, South: 180°, West: 270°
164    ///
165    /// # Examples
166    ///
167    /// ```
168    /// # use approx::assert_relative_eq;
169    /// use geo::{Haversine, Bearing};
170    /// use geo::Point;
171    ///
172    /// let origin = Point::new(9.0, 10.0);
173    /// let destination = Point::new(9.5, 10.1);
174    /// let bearing = Haversine.bearing(origin, destination);
175    /// // A little north of east
176    /// assert_relative_eq!(bearing, 78.47, epsilon = 1.0e-2);
177    /// ```
178    ///
179    /// # References
180    ///
181    /// Bullock, R.: Great Circle Distances and Bearings Between Two Locations, 2007.
182    /// (<https://dtcenter.org/met/users/docs/write_ups/gc_simple.pdf>)
183    ///
184    /// [great circle]: https://en.wikipedia.org/wiki/Great_circle
185    fn bearing(&self, origin: Point<F>, destination: Point<F>) -> F {
186        let three_sixty =
187            F::from(360.0).expect("Numeric type to be constructable from primitive 360");
188        let (lng_a, lat_a) = (origin.x().to_radians(), origin.y().to_radians());
189        let (lng_b, lat_b) = (destination.x().to_radians(), destination.y().to_radians());
190        let delta_lng = lng_b - lng_a;
191        let s = lat_b.cos() * delta_lng.sin();
192        let c = lat_a.cos() * lat_b.sin() - lat_a.sin() * lat_b.cos() * delta_lng.cos();
193
194        let degrees = F::atan2(s, c).to_degrees();
195        (degrees + three_sixty) % three_sixty
196    }
197}

199impl<F: CoordFloat + FromPrimitive> Destination<F> for HaversineMeasure {
200    /// Returns a new point having travelled the `distance` along a [great circle]
201    /// from the `origin` point with the given `bearing`.
202    ///
203    /// # Units
204    ///
205    /// - `origin`: Point where x/y are lon/lat degree coordinates
206    /// - `bearing`: degrees, where: North: 0°, East: 90°, South: 180°, West: 270°
207    /// - `distance`: meters
208    /// - returns: Point where x/y are lon/lat degree coordinates
209    ///
210    /// # Examples
211    ///
212    /// ```
213    /// # use approx::assert_relative_eq;
214    /// use geo::{Haversine, Destination};
215    /// use geo::Point;
216    ///
217    /// let origin = Point::new(9.177789688110352, 48.776781529534965);
218    /// let destination = Haversine.destination(origin, 45., 10000.);
219    /// assert_relative_eq!(Point::new(9.274409949623532, 48.84033274015048), destination);
220    /// ```
221    fn destination(&self, origin: Point<F>, bearing: F, meters: F) -> Point<F> {
222        let center_lng = origin.x().to_radians();
223        let center_lat = origin.y().to_radians();
224        let bearing_rad = bearing.to_radians();
225
226        let rad = meters / F::from(self.radius).unwrap();
227
228        let lat =
229            { center_lat.sin() * rad.cos() + center_lat.cos() * rad.sin() * bearing_rad.cos() }
230                .asin();
231        let lng = { bearing_rad.sin() * rad.sin() * center_lat.cos() }
232            .atan2(rad.cos() - center_lat.sin() * lat.sin())
233            + center_lng;
234
235        Point::new(normalize_longitude(lng.to_degrees()), lat.to_degrees())
236    }
237}

239impl<F: CoordFloat + FromPrimitive> Distance<F, Point<F>, Point<F>> for HaversineMeasure {
240    /// Determine the distance between two points using the [haversine formula].
241    ///
242    /// # Units
243    ///
244    /// - `origin`, `destination`: Points where x/y are lon/lat degree coordinates
245    /// - returns: meters
246    ///
247    /// # Examples
248    ///
249    /// ```
250    /// # use approx::assert_relative_eq;
251    /// use geo::{Haversine, Distance};
252    /// use geo::Point;
253    ///
254    /// let new_york_city = Point::new(-74.006f64, 40.7128f64);
255    /// let london = Point::new(-0.1278f64, 51.5074f64);
256    ///
257    /// let distance = Haversine.distance(new_york_city, london);
258    ///
259    /// assert_relative_eq!(
260    ///     5_570_230., // meters
261    ///     distance.round()
262    /// );
263    /// ```
264    ///
265    /// [haversine formula]: https://en.wikipedia.org/wiki/Haversine_formula
266    fn distance(&self, origin: Point<F>, destination: Point<F>) -> F {
267        let two = F::one() + F::one();
268        let theta1 = origin.y().to_radians();
269        let theta2 = destination.y().to_radians();
270        let delta_theta = (destination.y() - origin.y()).to_radians();
271        let delta_lambda = (destination.x() - origin.x()).to_radians();
272        let a = (delta_theta / two).sin().powi(2)
273            + theta1.cos() * theta2.cos() * (delta_lambda / two).sin().powi(2);
274        let c = two * a.sqrt().asin();
275        F::from(self.radius).unwrap() * c
276    }
277}

282impl<F: CoordFloat + FromPrimitive> InterpolatePoint<F> for HaversineMeasure {
283    /// Returns a new Point along a [great circle] between two existing points.
284    ///
285    /// # Examples
286    ///
287    /// ```
288    /// # use approx::assert_relative_eq;
289    /// use geo::{Haversine, InterpolatePoint};
290    /// use geo::Point;
291    ///
292    /// let p1 = Point::new(10.0, 20.0);
293    /// let p2 = Point::new(125.0, 25.0);
294    ///
295    /// let closer_to_p1 = Haversine.point_at_distance_between(p1, p2, 100_000.0);
296    /// assert_relative_eq!(closer_to_p1, Point::new(10.81, 20.49), epsilon = 1.0e-2);
297    ///
298    /// let closer_to_p2 = Haversine.point_at_distance_between(p1, p2, 10_000_000.0);
299    /// assert_relative_eq!(closer_to_p2, Point::new(112.33, 30.57), epsilon = 1.0e-2);
300    /// ```
301    ///
302    /// [great circle]: https://en.wikipedia.org/wiki/Great_circle
303    fn point_at_distance_between(
304        &self,
305        start: Point<F>,
306        end: Point<F>,
307        meters_from_start: F,
308    ) -> Point<F> {
309        let bearing = self.bearing(start, end);
310        self.destination(start, bearing, meters_from_start)
311    }
312
313    /// Returns a new Point along a [great circle] between two existing points.
314    ///
315    /// # Examples
316    ///
317    /// ```
318    /// # use approx::assert_relative_eq;
319    /// use geo::{Haversine, InterpolatePoint};
320    /// use geo::Point;
321    ///
322    /// let p1 = Point::new(10.0, 20.0);
323    /// let p2 = Point::new(125.0, 25.0);
324    ///
325    /// let closer_to_p1 = Haversine.point_at_ratio_between(p1, p2, 0.1);
326    /// assert_relative_eq!(closer_to_p1, Point::new(19.52, 25.27), epsilon = 1.0e-2);
327    ///
328    /// let closer_to_p2 = Haversine.point_at_ratio_between(p1, p2, 0.9);
329    /// assert_relative_eq!(closer_to_p2, Point::new(114.72, 29.65), epsilon = 1.0e-2);
330    ///
331    /// let midpoint = Haversine.point_at_ratio_between(p1, p2, 0.5);
332    /// assert_relative_eq!(midpoint, Point::new(65.87, 37.62), epsilon = 1.0e-2);
333    /// ```
334    ///
335    /// [great circle]: https://en.wikipedia.org/wiki/Great_circle
336    fn point_at_ratio_between(
337        &self,
338        start: Point<F>,
339        end: Point<F>,
340        ratio_from_start: F,
341    ) -> Point<F> {
342        if start == end || ratio_from_start == F::zero() {
343            return start;
344        }
345        if ratio_from_start == F::one() {
346            return end;
347        }
348        let calculation = HaversineIntermediateFillCalculation::new(start, end);
349        calculation.point_at_ratio(ratio_from_start)
350    }
351
352    /// Interpolates `Point`s along a [great circle] between `start` and `end`.
353    ///
354    /// As many points as necessary will be added such that the [haversine distance] between points
355    /// never exceeds `max_distance`. If the distance between start and end is less than
356    /// `max_distance`, no additional points will be included in the output.
357    ///
358    /// `include_ends`: Should the start and end points be included in the output?
359    ///
360    /// [great circle]: https://en.wikipedia.org/wiki/Great_circle
361    /// [haversine formula]: https://en.wikipedia.org/wiki/Haversine_formula
362    fn points_along_line(
363        &self,
364        start: Point<F>,
365        end: Point<F>,
366        max_distance: F,
367        include_ends: bool,
368    ) -> impl Iterator<Item = Point<F>> {
369        let calculation = HaversineIntermediateFillCalculation::new(start, end);
370        let HaversineIntermediateFillCalculation { d, .. } = calculation;
371
372        let total_distance = d * F::from(self.radius).unwrap();
373
374        if total_distance <= max_distance {
375            return if include_ends {
376                vec![start, end].into_iter()
377            } else {
378                vec![].into_iter()
379            };
380        }
381
382        let number_of_points = (total_distance / max_distance).ceil();
383        let interval = F::one() / number_of_points;
384
385        let mut current_step = interval;
386        let mut points = if include_ends { vec![start] } else { vec![] };
387
388        while current_step < F::one() {
389            let point = calculation.point_at_ratio(current_step);
390            points.push(point);
391            current_step = current_step + interval;
392        }
393
394        if include_ends {
395            points.push(end);
396        }
397
398        points.into_iter()
399    }
400}

402#[allow(clippy::many_single_char_names)]
403struct HaversineIntermediateFillCalculation<T> {
404    d: T,
405    n: T,
406    o: T,
407    p: T,
408    q: T,
409    r: T,
410    s: T,
411}
412
413impl<T: CoordFloat + FromPrimitive> HaversineIntermediateFillCalculation<T> {
414    #[allow(clippy::many_single_char_names)]
415    fn new(p1: Point<T>, p2: Point<T>) -> Self {
416        let one = T::one();
417        let two = one + one;
418
419        let lat1 = p1.y().to_radians();
420        let lon1 = p1.x().to_radians();
421        let lat2 = p2.y().to_radians();
422        let lon2 = p2.x().to_radians();
423
424        let (lat1_sin, lat1_cos) = lat1.sin_cos();
425        let (lat2_sin, lat2_cos) = lat2.sin_cos();
426        let (lon1_sin, lon1_cos) = lon1.sin_cos();
427        let (lon2_sin, lon2_cos) = lon2.sin_cos();
428
429        let m = lat1_cos * lat2_cos;
430
431        let n = lat1_cos * lon1_cos;
432        let o = lat2_cos * lon2_cos;
433        let p = lat1_cos * lon1_sin;
434        let q = lat2_cos * lon2_sin;
435
436        let k =
437            (((lat1 - lat2) / two).sin().powi(2) + m * ((lon1 - lon2) / two).sin().powi(2)).sqrt();
438
439        let d = two * k.asin();
440
441        Self {
442            d,
443            n,
444            o,
445            p,
446            q,
447            r: lat1_sin,
448            s: lat2_sin,
449        }
450    }
451
452    #[allow(clippy::many_single_char_names)]
453    fn point_at_ratio(&self, ratio_from_start: T) -> Point<T> {
454        let one = T::one();
455
456        let HaversineIntermediateFillCalculation {
457            d,
458            n,
459            o,
460            p,
461            q,
462            r,
463            s,
464        } = *self;
465
466        let a = ((one - ratio_from_start) * d).sin() / d.sin();
467        let b = (ratio_from_start * d).sin() / d.sin();
468
469        let x = a * n + b * o;
470        let y = a * p + b * q;
471        let z = a * r + b * s;
472
473        let lat = z.atan2(x.hypot(y));
474        let lon = y.atan2(x);
475
476        Point::new(lon.to_degrees(), lat.to_degrees())
477    }
478}
